\subsection{Addition}
\begin{exer}{2.2.1}{
    Prove Proposition 2.2.5 (Associativity of addition):
    For any natural numbers $a, b, c$, we have $(a + b) + c = a + (b + c)$.
}
    We induct on $a$, keeping $b$ and $c$ fixed.
    
    First, consider the base case $a = 0$.
    Then we have $(0 + b) + c = 0 + (b + c)$. By Lemma 2.2.2, we obtain $b + c = b + c$ as desired.
    \par
    Now we assume inductively that $(a + b) + c = a + (b + c)$;
    we now have to show $((\succr{a}) + b) + c = (\succr{a}) + (b+c)$ to close the induction.
    By applying Lemma 2.2.3 twice on the left-hand side, $((\succr{a}) + b) + c = (\succr{(a+b)}) + c = \succr{(a + b + c)}$.
    Similarly, applying Lemma 2.2.3 on the right-hand side yields $(\succr{a}) + (b+c) = \succr{(a + b + c)}$.
    Thus, both sides are equal, and we have closed the induction. \qed
\end{exer}


\begin{exer}{2.2.2}{
    Prove Lemma 2.2.10:
    Let $a$ be a positive number.
    Then there exists exactly one natural number $b$ such that $\succr{b} = a$.
}
    We induct on $a$.
    
    First, consider the base case $a = 1$.
    By Definition 2.1.3, $1 = \succr{0}$.
    Suppose there is another natural number $b$ such that $\succr{b} = 1$.
    However, by Axiom 2.4, if $\succr{b} = \succr{0} = 1$, then $b = 0$.
    Thus, there is exactly one natural number $b$ such that $\succr{b} = 1$.
    Hence, the base case is proven.

    Suppose inductively that there exists exactly one natural number $b$ such that $\succr{b} = a$.
    We now have to show there exists exactly one natural number $b'$ such that $\succr{b'} = \succr{a}$.
    As per Axiom 2.4, if $\succr{b'} = \succr{a'}$, then we must have $b' = a$, which is the only solution.
    This closes the induction. \qed
\end{exer}


\begin{exer}{2.2.3}{
    Prove Proposition 2.2.12 (Basic properties of order of natural numbers).
}
    Let $a, b, c$ be natural numbers. Then,
    \begin{enumerate}[label=(\alph*)]
        \item \textit{Reflexivity: $a \ge a$.}
        
            By Lemma 2.2.2, $a + 0 = a$. Thus, by Definition 2.2.11, $a \ge a$.

        \item \textit{Transitivity: If  $a \ge b$ and $b \ge c$, then $a \ge c$.}
        
            By Definition 2.2.11, write $a = b + m$, $b = c + n$, where $m$ and $n$ are natural numbers.
            Thus, $a = (c + n) + m = c + (n + m)$ by associativity of addition.
            By mathematical induction, it is trivial to prove that $n + m$ is a natural number.
            Therefore, $a \ge c$.

        \item \textit{Anti-symmetry: If $a \ge b$ and $b \ge a$, then $a = b$.}

            By Definition 2.2.11, if $a \ge b$, then for a natural number $m$, $a = b + m$.
            But also if $b \ge a$, then $b = a + n$, where $n$ is a natural number.
            Thus $a = (a + n) + m = a + (n + m)$.
            By Lemma 2.2.2 and Proposition 2.2.6 (cancellation law), $0 = n + m$.
            By Corollary 2.2.9, this results in $m = n = 0$.
            Thus $a = b + 0 = b$ as required.
            
        \item \textit{Preservation of order: $a \ge b$ if and only if $a + c \ge b + c$.}

            First, prove $a \ge b \implies a + c \ge b + c$.
            According to Definition 2.2.11, if $a \ge b$, then $a = b + m$, where $m$ is a natural number.
            Thus $a + c = (b + m) + c = (b + c) + m$, thus $a + c \ge b + c$.

            Conversely, consider $a + c = (b + c) + m$, where $m$ is a natural number.
            By Proposition 2.2.6 (cancellation law) and the associativity of addition, $a = b + m$, meaning $a \ge b$.

            Because $a \ge b \implies a + c \ge b + c$ and $a + c \ge b + c \implies a \ge b$, $a + c \ge b + c \iff a \ge b$.

        \item \textit{$a < b$ if and only if $\succr{a} < b$.}

            First, consider $a < b \implies \succr{a} \le b$.
            According to Definition 2.2.11, if $a < b$, then $b = a + m$, where $m$ is a natural number. 
            Note that $m \ne 0$ because we would otherwise obtain $a = b$, which is contradictory to the given condition $a < b$.
            Therefore, by Axiom 2.3, we have $m = \succr{n}$, where $n$ is a natural number.
            Thus, by the definition and commutativity of addition, $b = a + \succr{n} = \succr{(a + n)} = (\succr{a}) + n$.
            It follows that $\succr{a} \le b$.

            Now consider the converse.
            From $\succr{a} \le b$, we obtain $b = \succr{a} + p$, where $p$ is a natural number.
            Thus, by the definition and commutativity of addition, $b = \succr{a} + p = \succr{(a + p)} = a + \succr{p}$.
            According to Axiom 2.3, $0$ is not the successor of any natural number, so $\succr{p} \ne 0$.
            Hence, $a \le b$.

            Because $a < b \implies \succr{a} \le b$ and $\succr{a} \le b \implies a < b$, $a < b \iff \succr{a} \le {b}.$

        \item \textit{$a < b$ if and only if $b = a + d$ for some positive number $d$.}

            First, consider $a < b \implies b = a + d$. 
            From $a < b$ we know $a \ge b$, or $b = a + d$ for a natural number $d$.
            For the sake of contradiction, suppose $d = 0$.
            Therefore $a = b$, which contradicts $a < b$.
            Hence, $d$ is positive; thus, $a < b \implies b = a + d$ for some positive number $d$.

            Now consider the converse.
            From Definition 2.2.11, $b = a + d$ implies $b \ge a$, or $a \le b$.
            If $b = a$, by Proposition 2.2.6 (cancellation law), $d = 0$, which contradicts the given condition that $d$ is positive.
            Therefore, $a \ne b$, which implies $a < b$.

            Because the statement and the converse are both true, we can conclude that for some positive number $d$, $a < b$ if and only if $b = a + d$. \qed
            
    \end{enumerate}
\end{exer}

\begin{exer}{2.2.5}{
    Prove Proposition 2.2.14 (Strong principle of induction):
    Let $m_0$ be a natural number, and let $P(m)$ be a property pertaining to an arbitrary natural number $m$.
    Suppose that for each $m \ge m_0$, we have the following implication: 
    if $P(m')$ is true for all natural numbers $m_0 \le m' < m$, then $P(m)$ is also true.
    Then we can conclude that $P(m)$ is true for all natural numbers $m \ge m_0$.
}
    Let $Q(n)$ be the property such that $P(m)$ is true for all $m_0 \le m < n$.
    Note that the supposed implication becomes ``If $Q(n)$ is true, then $P(n)$ is also true.
    We induct on $n$.

    First, consider the base case $n = m_0$.
    In this case, $Q(m_0)$ is vacuously true as there is no such natural number $m$.
    Hence, the base case is proven.
    (Note that $Q(n)$ is also vacuously true for $n < m_0$.)

    Now, we assume inductively that $Q(n)$ is true; we will now prove $Q(\succr{n})$ is true.
    By the inductive hypothesis, $P(m)$ is true for all $m$ where $m_0 \le m < n$.
    According to the supposed implication, because $Q(n)$ is true, $P(n)$ is also true.
    Therefore, $P(m)$ is true for $m_0 \le m \le n$, which implies $Q(\succr{n})$ is also true.
    This closes the induction. \qed
\end{exer}


\begin{exer}{2.2.6}{
    Let $n$ be a natural number, and let $P(m)$ be a property pertaining to the natural numbers such that whenever $P(\succr{m})$ is true, then $P(m)$ is true.
    Suppose that $P(n)$ is also true.
    Prove that $P(m)$ is true for all natural numbers $m \le n$; this is known as the \emph{principle of backwards induction}.
}
    Write $Q(n)$ as the property pertaining to $n$ that ``If $P(n)$ is true, $P(m)$ is true for all natural numbers $m \le n$. We apply induction on $n$.

    Consider the base case $n = 0$. The statement $m \le n$ implies $0 = m + x$ for a natural number $x$.
    According to Corollary 2.2.9, this leads to $m = x = 0$.
    Thus, $Q(n)$ is equivalent to ``If $P(n)$ is true, $P(n)$ is true.", which is always true regardless of whether $P(n)$ is true or false.
    Hence, the base case is proven.

    Suppose inductively that $Q(n)$ is true; we now need to show $Q(\succr{n})$ is true.
    From the definition of $P(m)$, $P(\succr{n})$ being true implies $P(n)$ is true.
    By the inductive statement, $P(m)$ is true for all natural numbers $m \le n$.
    Thus if $P(\succr{n})$ is true, $P(m)$ is true for all $m \le \succr{n}$.
    This is precisely the definition of $Q(\succr{n})$, so $Q(\succr{n})$ is also true.
    This closes the induction. \qed
\end{exer}


\subsection{Multiplication}

\begin{exer}{2.3.1}{
    Prove Lemma 2.3.2 (Commutativity of multiplication):
    Let $n, m$ be natural numbers.
    Then $n \times m = m \times n$.
}
    First, we introduce a few lemmas.
    For all lemmas, we induct on $n$, keeping $m$ fixed (when it appears).
    \begin{lemma}
        For any natural number $n$, $n \times 0 = n$.
    \end{lemma}
    \begin{prf}
        In the base case $n = 0$, $0 \times 0 = 0$ is true by the definition of multiplication, and $0$ is a natural number.
        Now suppose inductively that $n \times 0 = 0$; we will now prove $(\succr{n}) \times 0 = 0$.
        By the definition of multiplication, $(\succr{n}) \times 0 = (n \times 0) + 0 = 0 + 0 = 0$.
    \end{prf}
    \begin{lemma}
        For any natural numbers $n$ and $m$, $n \times (\succr{m}) = (n \times m) + n$.
    \end{lemma}
    \begin{prf}
        In the base case $n = 0$, $0 \times \succr{m} = 0$ by the definition of multiplication. 
        Also, $(0 \times m) + 0 = 0$.
        Thus, the left-hand side is true.
        
        Now suppose inductively that $n \times (\succr{m}) = (n \times m) + n$;
        we must now show $(\succr{n}) \times (\succr{m}) = ((\succr{n}) \times m) + (\succr{n})$.
        We start on the left-hand side.
        \begin{align*}
              & (\succr{n}) \times (\succr{m})                                       \\
            = & n \times (\succr{m}) + \succr{m}     & \text{(Definition 2.3.1)}     \\
            = & (n \times m) + n + \succr{m}         & \text{(Inductive hypothesis)} \\
            = & (n \times m) + m + \succr{n}         & \text{(Addition properties)}  \\
            = & ((\succr{n}) \times m) + (\succr{n}) & \text{(Definition 2.3.1)}
        \end{align*}
        This closes the induction.
    \end{prf}

    We can now prove the main statement $n \times m = m \times n$.
    Consider the base case $n = 0$. Thus, $0 \times m = m \times 0 = 0$.
    Hence, the base case is true.
    
    Now suppose inductively that $n \times m = m \times n$;
    we will now prove $(\succr{n}) \times m = m \times (\succr{n})$.
    By the definition of multiplication, $(\succr{n}) \times m = (n \times m) + m$.
    From the second lemma, the right-hand side is also equivalent to $(n \times m) + m$.
    Because both sides are equal, the statement is true. This closes the induction. \qed
\end{exer}


\begin{exer}{2.3.2}{
    Prove Lemma 2.3.3 (Positive natural numbers have no zero divisors):
    Let $m, n$ be natural numbers.
    Then $n \times m = 0$ if and only if at least one of $n, m$ is equal to zero.
    In particular, if $n$ and $m$ are both positive, then $nm$ is also positive.
}
    We will prove the second statement first.
    For the sake of contradiction, suppose there exist positive numbers $m, n$ such that $mn = 0$.
    Because $m, n$ are positive numbers, write $m = \succr{p}, n = \succr{q}$, thus $(\succr{q}) \times (\succr{p})  = 0$.
    From the definition of multiplication, $(q \times (\succr{p})) + (\succr{p}) = 0$.
    According to Corollary 2.2.9, this implies $q \times (\succr{p}) = \succr{p} = 0$,
    which is contradictory as Axiom 2.3 states $0$ is not the successor of any natural number.
    Hence, if $m, n$ are both positive, then $mn$ is positive.

    Now consider the first statement.
    The left-to-right implication is simply the contrapositive of the second statement, which is true.
    Observe the right-to-left implication.
    If $n = 0$, then $0 \times m = 0$ by the definition of multiplication.
    If $m = 0$, then by commutativity of multiplication, $m \times 0 = 0 \times m = 0$.
    Hence, the statements are proven as desired. \qed
\end{exer}


\begin{exer}{2.3.3}{
    Prove Proposition 2.3.5 (Associativity of multiplication):
    For any natural numbers $a, b, c$, we have $(a \times b) \times c = a \times (b \times c)$.
}
    We induct on $a$, keeping $b$ and $c$ fixed.
    
    First, consider the base case $a = 0$.
    According to the definition of multiplication, the left-hand side leads to $(0 \times b) \times c = 0 \times c = 0$.
    Similarly, the right-hand side results in $0 \times (b \times c) = 0$.
    Because both sides are equal, our base case is proven.

    Now we assume inductively that $(a \times b) \times c = a \times (b \times c)$;
    we need to show that $((a+1) \times b) \times c = (a+1)(b \times c)$.
    By the distributive law, the left-hand side results in $((a + 1) \times b) \times c = (a \times b + b) \times c = (a \times b) \times c + b \times c$.
    The right-hand side produces $(a + 1) \times (b \times c) = a \times (b \times c) + b \times c$, which, by the inductive hypothesis, is equal to $(a \times b) \times c + b \times c$.
    Thus, both sides are equal, and this closes the induction. \qed
\end{exer}

\begin{exer}{2.3.5}{
    Prove Proposition 2.3.9 (Euclidean algorithm):
    Let $n$ be a natural number, and let $q$ be a positive number.
    Then there exist natural numbers $m, r$ such that $0 \le r < q$ and $n = mq + r$.
}
    We induct on $n$, fixing $q$.

    First, observe the base case $n = 0$.
    There is an obvious solution of $m = 0, r = 0$.
    Hence, the base case is proven.

    Now, suppose inductively that there exist natural numbers $m, r$ such that $0 \le r < q$ and $n = mq + r$;
    we now have to show there exists natural numbers $m', r'$ such that $0 \le r' < q$ and $n+1 = m'q + r'$.
    By the induction hypothesis, $n + 1 = mq + r + 1$.
    Since $r < q$, $r + 1 \le q$.
    Thus, we have two cases:
    \begin{enumerate}[label=(\alph*), topsep=0mm]
        \item $r + 1 < q$: In this case, we can have $n + 1 = mq + (r + 1)$ with $0 \le r + 1 < q$. Thus, we can have $m' = m$, $r' = r + 1$.
        \item $r + 1 = q$: In this case, we have $n + 1 = mq + q = (m + 1)q$ by the distributive law. Therefore, $m' = m + 1, r' = 0$.
    \end{enumerate}
    This closes the induction. \qed
\end{exer}

\begin{exer}{3.1.1}{
    Show that the definition of equality in Definition 3.1.4 is reflexive, symmetric, and transitive.
}
    Let $A$, $B$, and $C$ be sets.
    \begin{enumerate}[topsep=0mm, label=(\alph*)]
        \item \textit{Reflexivity: $A = A$}

            For every element $x$ of $A$, $x$ is clearly an element of $A$.
            Thus by Definition 3.1.4, $A = A$.

        \item \textit{Symmetry: If $A = B$, then $B = A$.}

            Suppose $A = B$.
            Then for every element $x \in B$, $x \in A$, and for every element $y \in A$, $y \in B$.
            Thus $B = A$.

        \item \textit{Transitivity: If $A = B$ and $B = C$, then $A = C$.}
        
            Suppose $A = B$ and $B = C$.
            Thus, for an arbitrary element $x \in A$, $x \in B$ because $A = B$,
            and since $B = C$, we see that $x \in C$.
            Similarly, for an arbitrary element $y \in C$, it follows that $y \in B$, and then $y \in A$.
            Hence, by Definition 3.1.4, it follows that $A = C$ if $A = B$ and $B = C$. \qed
    \end{enumerate}
\end{exer}


\begin{exer}{3.1.2}{
    Using only Definition 3.1.4, Axiom 3.1, Axiom 3.2, and Axiom 3.3, prove that the sets $\varnothing$, $\{ \varnothing \}$, $\{ \{ \varnothing \} \}$, and $\{ \varnothing, \{ \varnothing \} \}$ are all distinct (i.e., no two of them are equal to each other).
}
    First, let's show that $\varnothing$ is different from every other set.
    The empty set $\varnothing$ has no elements.
    However, $\varnothing$ is an element of $\{ \varnothing \}$ and $\{\varnothing, \{ \varnothing \} \}$, and $\{ \varnothing \}$ is an element of $\{ \{ \varnothing \} \}$.
    Thus, $\varnothing$ is different from the other sets.

    Now observe the set $\{ \varnothing \}$.
    According to Axiom 3.3, $x \in \{ \varnothing \}$ if and only if $x = \varnothing$.
    However, $\{ \varnothing \}$ is an element of both $\{ \{ \varnothing \} \}$ and $\{ \varnothing, \{ \varnothing \} \}$.
    Hence, $\{ \varnothing \}$ is different from the other sets.

    We now need to show that $\{ \{ \varnothing \} \} \ne \{ \varnothing, \{ \varnothing \} \}$.
    Notice $\varnothing \notin \{ \{ \varnothing \} \}$ but $\varnothing \in \{ \varnothing, \{ \varnothing \} \}$.
    Therefore, $\{ \{ \varnothing \} \}$ and $\{ \varnothing, \{ \varnothing \} \}$ are distinct sets too. We conclude that the sets $\varnothing$, $\{ \varnothing \}$, $\{ \{ \varnothing \} \}$, and $\{ \varnothing, \{ \varnothing \} \}$ are all distinct. \qed
\end{exer}


\begin{exer}{3.1.3}{
    Prove the remaining claims in Lemma 3.1.13.
}
    \begin{enumerate}[topsep=0mm, label=(\alph*)]
        \item \textit{If $a$ and $b$ are objects, then $\{ a, b \} = \{ a \} \cup \{ b \}$.}

            By Axiom 3.4, for an object $x$, $x \in \{ a \} \cup \{ b \} \iff (x \in \{ a \}  \text{ or } x \in \{ b \})$.
            Also, Axiom 3.3 states that $x \in \{ a \} \iff x = a$.
            Similarly, $x \in \{ b \} \iff x = b$.
            Axiom 3.3 also states that $x = a$ or $x = b$ if and only if $x \in \{ a, b \}$.
            Therefore, if $x \in \{ a \} \cup \{ b \}$, $x \in \{ a, b \}$.
            According to Definition 3.1.4, this implies $\{ a, b \} = \{ a \} \cup \{ b \}$.

        \item \textit{(Commutativity of the union operation) $A \cup B = B \cup A$.}

            For any object $x$, $x \in A \cup B \iff (x \in A \text{ or } x \in B)$.
            If $x \in A$, then by Axiom 3.4, then $x \in B \cup A$.
            Similarly, if $x \in B$, then $x \in B \cup A$.
            We can show in a similar fashion that $x \in B \cup A \implies x \in A \cup B$.
            Hence, by Definition 3.1.4, $A \cup B = B \cup A$.

        \item \textit{$A \cup A = A \cup \varnothing = \varnothing \cup A = A$.}

            First, by Axiom 3.4, for an object $x$, $x \in A \cup A \iff (x \in A \text{ or } x \in A)$.
            But, $(x \in A \text{ or } x \in A)$ is equivalent to $x \in A$.
            So, $x \in A \cup A \iff x \in A$, thus $A \cup A = A$.

            Now consider $A \cup \varnothing$.
            From Axiom 3.4, for an object $y$, $y \in A \cup \varnothing \iff (y \in A \text{ or } y \in \varnothing)$.
            However, Axiom 3.2 states that for an arbitrary $y$, $y \not in \varnothing$.
            Therefore, $y \in A \cup \varnothing \iff y \in A$, which implies $A \cup \varnothing = A$.

            As we have proven the union operation is commutative, $A \cup \varnothing = \varnothing \cup A = A$. \qed
            
    \end{enumerate}
\end{exer}


\begin{exer}{3.1.4}{
    Prove the remaining claims in Proposition 3.1.18.
}
    \begin{enumerate}[topsep=0mm, label=(\alph*)]
        \item \textit{If $A \subseteq B$ and $B \subseteq A$, then $A = B$.}

            According to Definition 3.1.15, because $A \subseteq B$, for any object $x$, $x \in A \implies x \in B$.
            Similarly, because $B \subseteq A$, $x \in B \implies x \in A$.
            Thus, $x \in A \iff x \in B$, which implies $A = B$.

        \item \textit{If $A \subsetneq B$ and $B \subsetneq C$, then $A \subsetneq C$.}

            Let $x$ be an arbitrary element of $A$.
            Since $A \subsetneq B$ and $B \subsetneq C$, it follows that $x \in B$, and thus $x \in C$.
            This shows that $A \subseteq C$.
            However, because $A \subsetneq B$, there exists an element $y \in B$ with $y \notin A$.
            Since $B \subsetneq C$, $y \in C$.
            Thus, we have an element $y$ where $y \in C$ and $y \notin A$.
            This shows that $A \ne C$.
            Hence, because $A \subseteq C$ and $A \ne C$, we can conclude that $A \subsetneq C$. \qed
    \end{enumerate}
\end{exer}


\begin{exer}{3.1.5}{
    Let $A, B$ be sets.
    Show that the three statements $A \subseteq B$, $A \cup B = B$, $A \cap B = A$ are logically equivalent (any one of them implies the other two).
}
    First, we show that $A \subseteq B \implies A \cup B = B$.
    To prove $A \cup B = B$, we need to prove that if $A \subseteq B$, then $x \in A \cup B \iff x \in B$ for any object $x$.
    Clearly, $x \in B \implies x \in A \cup B$, so we only need to prove $x \in A \cup B \implies x \in B$.
    Because $A \subseteq B$, we know $x \in A \implies x \in B$.
    Thus, $x \in A \cup B$ implies $x \in A$ or $x \in B$, which is logically equivalent to $x \in B$.
    Therefore, $A \subseteq B \implies A \cup B = B$.

    Now consider $A \cup B = B \implies A \cap B = A$.
    To prove $A \cap B = A$, we need to show that if $A \cup B = B$, then $x \in A \cap B \iff x \in A$.
    Proving $x \in A \cap B \implies x \in A$ is trivial based on Definition 3.1.23.
    Consider the converse implication $x \in A \implies x \in A \cap B$.
    Suppose $x \in A$.
    This implies $x \in A \cup B$, and because $A \cup B = B$, $x \in B$.
    By Definition 3.1.23, $x \in A$ and $x \in B \iff x \in A \cap B$.
    Therefore, $A \cup B = B \implies A \cap B = A$.

    Lastly, we need to prove $A \cap B = A \implies A \subseteq B$.
    Because $A \cap B = A$, for all $x \in A, x \in A \cap B$.
    However, $x \in A \cap B \iff x \in A \text{ and } x \in B$.
    Therefore, $x \in A \implies x \in B$. \qed
\end{exer}


\begin{exer}{3.1.6}{
    Prove Proposition 3.1.28 (Sets form a boolean algebra).
}
    Let $A, B, C$ be sets, and let $X$ be a set containing $A, B, C$ as subsets.
    \begin{enumerate}[topsep=0mm, label=(\alph*)]
        \item \textit{(Minimal element) We have $A \cup \varnothing = A$ and $A \cap \varnothing = \varnothing$.}

            Consider $A \cup \varnothing$.
            From Axiom 3.4, for an object $y$, $y \in A \cup \varnothing \iff (y \in A \text{ or } y \in \varnothing)$.
            However, Axiom 3.2 states that for an arbitrary $y$, $y \notin \varnothing$.
            Therefore, $y \in A \cup \varnothing \iff y \in A$, which implies $A \cup \varnothing = A$.

            Now consider $A \cap \varnothing$.
            For an object $y$, $y \in A \cap \varnothing \iff (y \in A \text{ or } y \in \varnothing)$.
            However, Axiom 3.2 states that for an arbitrary $y$, $y \notin \varnothing$.
            Therefore, no elements fulfil such conditions, or $A \cap \varnothing = \varnothing$.
        
        \item \textit{(Maximal element) We have $A \cup X = X$ and $A \cap X = A$.}

            Consider $A \cup X$.
            From Axiom 3.4, for an object $y$, $y \in A \cup X \iff y \in A \text{ or } y \in X$.
            However, if $y \in A$, then $y \in X$ because $A \subseteq X$.
            Thus, $y \in A \cup X \iff y \in A$, or $A \cup X = X$

            Now consider $A \cap X = A$.
            From Definition 3.1.23, $y \in A \cap X \iff y \in A \text{ and } y \in X$.
            However, if $y \in A$, then $y \in X$ because $A \subseteq X$.
            Thus, $y \in A \cap X \iff y \in A$, or $A \cap X = A$.
            
        \item \textit{(Identitiy) We have $A \cap A= A$ and $A \cup A = A$.}

            First, consider $A \cap A = A$.
            Definition 3.1.23 states that if an arbitrary object $y \in A \cap A$, then $y \in A$ and $y \in A$.
            Thus, $y \in A \cap A \iff y \in A$, or $A \cap A = A$ as required.
            Similarly, for $A \cup A = A$, Axiom 3.4 states that if an arbitrary object $y \in A \cup A$, then $y \in A$ or $y \in A$.
            This leads to $y \in A \cup A \iff y \in A$, or $A \cup A = A$ as required.

        \item \textit{(Commutativity) We have $A \cup B = B \cup A$ and $A \cap B = B \cap A$.}

            Commutativity of the union operation, or $A \cup B = B \cup A$, has been proven in Exercise 3.1.3(b).
            We will only show that the intersection operation is commutative, or $A \cap B = B \cap A$.
            Let $y$ be an object such that $y \in A \cap B$.
            Then $y \in A$ and $y \in B$.
            This is logically equivalent to $y \in B$ and $y \in A$.
            The converse is proven in a similar fashion; hence, $y \in A \cap B \iff y \in B \cup A$, or $A \cap B = B \cap A$.
        
        \item \textit{(Associativity) We have $(A \cup B) \cup C = A \cup (B \cup C)$and $(A \cap B) \cap C = A \cap (B \cap C)$.}

            The proof of the associativity of the union operation can be found in Lemma 3.1.13.
            We will only prove the associativity of the intersection operation.
            By Definition 3.1.14, we need to prove that all elements of $(A \cap B) \cap C$ are elements of $A \cap (B \cap C)$.
            Suppose $y$ is an arbitrary element of $(A \cap B) \cap C$.
            Thus, by Definition 3.1.23, $y$ is both an element of $(A \cap B)$ and $C$.
            Because $y \in A \cap B$, $y \in A$ and $y \in B$.
            Since $y \in B$ and $y \in C$, then $y \in B \cap C$.
            Also, as $y \in A$, we can conclude that $y \in A \cap (B \cap C)$.
            The converse is proven similarly.
            Hence, $(A \cap B) \cap C = A \cap (B \cap C)$ as desired.
        
        \item \textit{(Distributivity) We have $A \cap (B \cup C) = (A \cap B) \cup (A \cap C)$ and $A \cup (B \cap C)=(A \cup B) \cap (A \cup C)$.}

            First consider $A \cap (B \cup C) = (A \cap B) \cup (A \cap C)$.
            Suppose $x \in A \cap (B \cup C)$.
            Then, $x \in A$ and $x \in B \cup C$.
            So $x \in B$ or $x \in C$.
            We will now split into two cases.
            If $x \in B$, then $x \in A \cap B$, thus $x \in (A \cap B) \cup (A \cap C)$.
            Similarly, if $x \in C$, then $x \in A \cap C$, thus $x \in (A \cap B) \cup (A \cap C)$.
            Thus, $x \in A \cap (B \cup C) \implies x \in (A \cap B) \cup (A \cap C)$.
            
            For the converse, suppose instead $x \in (A \cap B) \cup (A \cap C)$.
            Then, $x \in A \cap B$ or $x \in(A \cap C) $.
            If $x \in A \cap B$, then $x \in A$ and $x \in B$.
            Because $x \in B$, we know $x \in B \cup C$.
            Therefore, $x \in A \cap (B \cup C)$.
            In a similar fashion, if $x \in A \cap C$, we can conclude that $x \in A \cap (B \cup C)$.
            Hence, $A \cap (B \cup C) = (A \cap B) \cup (A \cap C)$ as required.

            We will now consider $A \cup (B \cap C) = (A \cup B) \cap (A \cup C)$.
            Suppose $x$ is an element of $A \cup (B \cap C)$.
            We know that $x \in A$ or $x \in (B \cap C)$.
            We will now split into two cases.
            If $x \in A$, then $x \in A \cup B$ and $x \in A \cup C$.
            If instead $x \in B \cap C$, then $x \in B$ and $x \in C$, so $x \in A \cup B$ and $x \in A \cup C$ as well.
            In both cases, we can conclude that $x \in (A \cup B) \cap (A \cup C)$.
        
        \item \textit{(Partition) We have $A \cup (X \setminus A) = X$ and $A \cap (X \setminus A) = \varnothing$.}

            We will first consider $A \cup (X \setminus A)$.
            We know that $x \in A$ or $x \in X \setminus A$.
            If $x \in A$, then $x \in X$ because $A \subseteq X$.
            If $x \in X \setminus A$ instead, then $x \in X$ and $x \notin A$.
            Either way, we obtain $x \in X$, thus $x \in A \cup (X \setminus A) \implies x \in X$.
            
            Now consider the converse.
            Suppose instead $x \in X$.
            If $x \in A$, then $x \in A \cup (X \setminus A)$.
            If instead $x \notin A$, then $x \in X \setminus A$ because $x \in X$.
            Either way, $x \in X \implies x \in A \cup (X \setminus A)$.
            Hence, $A \cup (X \setminus A) = X$ as desired.

            Consider instead $A \cap (X \setminus A)$.
            Suppose $x \in A \cap (X \setminus A)$.
            Then, $x \in A$ and $x \in X \setminus A$.
            Because $x \in X \setminus A$, $x \notin A$, which contradicts $x \in A$ obtained earlier.
            Thus, no element $x$ fulfills $x \in A \cap (X \setminus A)$, or $x \in \varnothing$.
            Hence $A \cap (X \setminus A) = \varnothing$.
        
        \item \textit{(De Morgan Laws) We have $X \setminus (A \cup B) = (X \setminus A) \cap (X \setminus B)$ and $X \setminus (A \cap B) = (X \setminus A) \cup (X \setminus B)$.}

            First prove $X \setminus (A \cup B) = (X \setminus A) \cap (X \setminus B)$.
            Suppose $x \in X \setminus (A \cup B)$.
            Then, $x \in X$ and $x \notin A \cup B$, or $x \notin A$ and $x \notin B$.
            Thus, we obtain $x \in X \setminus A$ and $x \in X \setminus B$.
            This leads to $x \in (X \setminus A) \cap (X \setminus B)$.
            
            Now consider the converse.
            Suppose instead $x \in (X \setminus A) \cap (X \setminus B)$.
            Because $x \in X \setminus A$ and $X \setminus B$, $x \in X$, $x \notin A$ and $x \notin B$.
            Thus, $x \notin A \cup B$, and $x \in X \setminus (A \cup B)$.
            Hence, $X \setminus (A \cup B) = (X \setminus A) \cap (X \setminus B)$.

            We will now show that $X \setminus (A \cap B) = (X \setminus A) \cup (X \setminus B)$.
            Suppose $x \in X \setminus (A \cap B)$.
            Then, $x \in X$ and $x \notin A \cap B$.
            Because $x \notin A \cap B$, $x \notin A$ or $x \notin B$.
            If $x \notin A$, then $x \notin A$ or $x \notin B$ since the former statement is true.
            Thus, $x \in (X \setminus A)$ or $x \in (X \setminus B)$, or $x \in (X \setminus A) \cup (X \setminus B)$.

            Consider instead the converse.
            Suppose instead $x \in (X \setminus A) \cup (X \setminus B)$.
            Then, $x \in (X \setminus A)$ or $x \in (X \setminus B)$.
            If $x \in (X \setminus A)$, then $x \in X$ and $x \notin A$.
            Because $x \notin A$, $x \notin A \cap B$.
            Thus, $x \in X \setminus (A \cap B)$ because $x \in X$.
            Similarly, if $x \in (X \setminus B)$, $x \in X \setminus (A \cap B)$.
            Hence, we obtain $X \setminus (A \cap B) = (X \setminus A) \cup (X \setminus B)$. \qed

    \end{enumerate}
\end{exer}


\begin{exer}{3.1.7}{
    Let $A, B, C$ be sets.
    Show that $A \cap B \subseteq A$ and $A \cap B \subseteq B$.
    Furthermore, show that $C \subseteq A$ and $C\subseteq B$ if and only if $C \subseteq A \cap B$.
    In a similar spirit, show that $A \subseteq A \cup B$ and $B \subseteq A \cup B$,
    and furthermore that $A \subseteq C$ and $B \subseteq C$ if and only if $A \cup B \subseteq C$.
}
    \begin{enumerate}[topsep=0mm, label=(\alph*)]
        \item \textit{Show that $A \cap B \subseteq A$ and $A \cup B \subseteq B$.}
        
            From Definition 3.1.23, for all objects $x \in A \cap B$, $x \in A$ and $x \in B$.
            Thus, $x \in A \cap B \implies x \in A$, which is equivalent to $A \cap B \subseteq A$.
            Similarly, using the commutativity of the union operation, $A \cup B \subseteq B$.

        \item \textit{Show that $C \subseteq A$ and $C\subseteq B$ if and only if $C \subseteq A \cap B$.}
        
            Suppose $x \in C$.
            We will first prove the implication.
            If $C \subseteq A$ and $C \subseteq B$, then $x \in A$ and $x \in B$.
            Therefore, $x \in A \cap B$, and $C \subseteq A \cap B$.
            Now consider the converse.
            If $C \subseteq A \cap B$, then $x \in A \cap B$.
            Thus, $x \in A$ and $x \in B$, or $C \subseteq A$ and $C \subseteq B$.

        \item \textit{Show that $A \subseteq A \cup B$ and $B \subseteq A \cup B$.}

            For all objects $x \in A$, $x \in A \cup B$, thus $A \subseteq A \cup B$.
            Similarly, for all objects $x \in B$, $x \in A \cup B$, thus $B \subseteq A \cup B$.

        \item \textit{Show that $A \subseteq C$ and $B \subseteq C$ if and only if $A \cup B \subseteq C$.}

            We will first prove the implication.
            Suppose $x \in A \cup B$. Then, $x \in A$ or $x \in B$.
            If $x \in A$, then $x \in C$ because $A \subseteq C$.
            Similarly, if $x \in B$, then $x \in C$.
            Therefore, $A \cup B \subseteq C$.
            
            Now consider the converse.
            Suppose instead $x \in A$; thus $x \in A \cup B$.
            Because $A \cup B \subseteq C$, $x \in A \cup B \implies x \in C$.
            Thus $x \in A \implies x \in C$, or $A \subseteq C$.
            Similarly, by swapping the roles of $A$ and $B$ in the above argument, $B \subseteq C$.
            Hence, the statement is proven. \qed
            
    \end{enumerate}
\end{exer}


\begin{exer}{3.1.8}{
    Let $A, B$ be sets.
    Prove the \emph{absorption laws} $A \cap (A \cup B) = A$ and $A \cup (A \cap B) = A$.
}
    First, consider $A \cap (A \cup B) = A$.
    Suppose $x$ is an element of $A \cap (A \cup B)$; thus $x \in A$ and $x \in A \cup B$.
    In particular, because $x \in A$, we obtain $A \cap (A \cup B) \subseteq A$.
    Instead, let $x \in A$; thus $x \in A \cup B$.
    This leads to $x \in A \cap (A \cup B)$, so $A \subseteq A \cap (A \cup B)$.
    Hence, $A \cap (A \cup B) = A$.

    Now consider $A \cup (A \cap B) = A$.
    Using the distributivity of the union operation proven in Proposition 3.1.28 (f), the left-hand side is equivalent to $(A \cup A) \cap (A \cup B) = A \cap (A \cup B)$.
    From the previous results, this is equal to $A$.
    Therefore, we obtain $A \cup (A \cap B) = A$ as desired. \qed    
\end{exer}


\begin{exer}{3.1.9}{
    Let $A, B, X$ be sets such that $A \cup B = X$ and $A \cap B = \varnothing$.
    Show that $A = X \setminus B$ and $B = X \setminus A$.
}
    Suppose $x \in A$.
    From $A \cup B = X$, we know $x \in A \cup B$, and thus $x \in X$.
    Also, from $A \cap B = \varnothing$, $A$ and $B$ are disjoint, therefore $x \notin B$.
    Thus, because $x \in X$ and $x \notin B$, $x \in X \setminus B$.
    Suppose instead that $x \in X \setminus B$.
    This implies $x \in X$ and $x \notin B$.
    Because $x \notin B$ and $x \in A \cup B$, $x \in A$.
    Hence, $A = X \setminus B$.
    In a similar fashion, we can swap the roles of $A$ and $B$ in the above proof to show $B = X \setminus A$. \qed
\end{exer}


\begin{exer}{3.1.10}{
    Let $A$ and $B$ be sets.
    Show that the three sets $A \setminus B$, $A \cap B$ and $B \setminus A$ are disjoint, and that their union is $A \cup B$.
}
    To show the three sets are mutually disjoint, we prove that the intersections of any two of the sets have no elements.
    Elements in $A \setminus B$ are in $A$ and not in $B$.
    However, elements in $A \cap B$ are in both $A$ and $B$,
    and elements in $B \setminus A$ are in set $B$ and not $A$.
    Because an object cannot both be in a set and not in a set, all three intersections are empty sets.

    We will now prove that the union of the three sets is $A \cup B$.
    Consider the following cases for an object $x$:
    \begin{enumerate}[topsep=0mm, label=(\alph*)]
        \item $x \in A$, $x \in B$: $x \in A \cap B$;
        \item $x \in A$, $x \notin B$: $x \in A \setminus B$;
        \item $x \notin A, x \in B$: $x \in B \setminus A$,
        \item $x \notin A, x \notin B$: $x$ is not an element of any of the sets, and is therefore also not in the union.
    \end{enumerate}
    Thus, if $x \in A$ or $x \in B$, $x$ is an element of exactly one of the sets.
    This is logically equivalent to saying $A \cup B$ is the union of the three sets as per Axiom 3.4. \qed
\end{exer}


\begin{exer}{3.1.11}{Show that the axiom of replacement implies the axiom of specification.}
    Let $A$ be a set, and let $P(x)$ be a property pertaining to $x$.
    To prove the axiom of specification, we need to construct the set $\{x \in A: P(x) \}$ using only the axiom of replacement.
    Let $Q(x, y)$ be the property that $y = x$ and $P(x)$ is true.
    To use the axiom of replacement, for each $x \in A$, there must be at most one $y$ such that $Q(x, y)$ is true.
    Indeed, for each $x \in A$, there is only one $y$ such that $y = x$.
    Therefore, there is at most one $y$ such that $Q(x, y)$ is true.
    We can now show that there exists a set $\{y : Q(x, y) \text{ for some } x \in A \}$ such that for an object $z$,
    \begin{align*}
             & z \in \{y : Q(x, y) \text{ for some } x \in A \} \\
        \iff & Q(x, z) \text{ is true for some } x \in A \\
        \iff & x = z \text{ and } P(x) \text{ is true for some } x \in A \\
        \iff & P(x) \text{ for some } x \in A \\
        \iff & P(z) \text{ for some } z \in A.
    \end{align*}
    Thus, we can construct the set $\{x \in A: P(x) \}$, which means we have proven the axiom of specification. \qed
\end{exer}


\subsection{Russell's Paradox}

\begin{exer}{3.2.1}{
    Show that the universal specification axiom, Axiom 3.8, if assumed to be true, would imply Axioms 3.2, 3.3, 3.4, 3.5, and 3.6.
    (If we assume that all natural numbers are objects, we also obtain Axiom 3.7.)
}
    Recall that Axiom 3.8 states that if for every object $x$ we have a property $P(x)$ pertaining to $x$, then there exists a set $\{ x : P(x) \}$ such that for every object $y$, $y \in \{x : P(x) \} \iff P(y)$.
    
    \begin{enumerate}[topsep=0mm, label=(\alph*)]
        \item \textit{Axiom 3.2: Empty set.}
        
            Let $P(x)$ be a property pertaining to an object $x$ that is always false.
            From Axiom 3.8, the set $\{ x : P(x) \}$ exists.
            However, for any object $y$, $P(y)$ is false.
            It follows that $y \in \{ x : P(x) \}$ is false for all objects $y$.
            Therefore, the set $\{ x : P(x) \}$ has no elements.
            This is precisely the claim of Axiom 3.2.
            
        \item \textit{Axiom 3.3: Singleton sets and pair sets.}
        
            Let $a$ be an arbitrary object and $P(x)$ be the property pertaining to $x$ such that $x = a$.
            From Axiom 3.8, the set $\{ x : x = a \}$ exists, and it has exactly one element $a$.
            Thus, the singleton set $\{ a \}$ exists.
                   
            Suppose instead $a$ and $b$ are arbitrary objects, and let $Q(x)$ be the property pertaining to $x$ such that $x = a$ or $x = b$.
            From Axiom 3.8, the set $\{ x : x = a \text{ or } x = b\}$ exists, which has exactly two elements $a$ and $b$.
            Thus, the pair set $\{ a, b \}$ exists.
            
        \item \textit{Axiom 3.4: Pairwise union.}
        
            Let $A$ and $B$ be sets, and $x$ be an object.
            Write $P(x)$ as the property pertaining to $x$ that $x \in A$ or $x \in B$.
            Thus, there exists a set $\{x : x \in A \text{ or } x \in B \}$.        
            We denote this set as the union of sets $A$ and $B$, or $A \cup B$.
            For any object $y$, $(y \in A \text{ or } y \in B) \iff y \in A \cup B$.
            This is precisely the claim of Axiom 3.4.
            
        \item \textit{Axiom 3.5: Axiom of specification.}
        
            Let $A$ be a set, and let $P(x)$ be a property pertaining to an object $x \in A$.
            Let $Q(x)$ be the property that $P(x)$ is true and $x \in A$.
            By Axiom 3.8, the set $\{ x : Q(x) \}$ exists.
            For any object $y$, $y \in \{ x : Q(x) \} \iff (y \in A \text{ and } P(y))$.
            This is exactly the claim of Axiom 3.5.
            
        \item \textit{Axiom 3.6: Replacement.}
        
            Let $A$ be a set, and $P(x, y)$ be a property pertaining to $x$ and $y$.
            Let $Q(y)$ be a property pertaining to $y$ that $P(x, y)$ is true for some $x \in A$.
            According to Axiom 3.8, there exists a set $\{ y : P(x, y) \text{ for some } x \in A \}$, such that for any object $z$, $z \in \{ y : P(x, y) \text{ for some } x \in A \} \iff P(x, z) \text{ for some } x \in A$.
            This is exactly the claim of Axiom 3.6.
            
        \item \textit{Axiom 3.7: Infinity.}
        
            Suppose all natural numbers are objects.          
            Let $P(x)$ be the statement that $x$ is a natural number.
            Then the set $\{ x : P(x) \}$ exists.
            We denote this set as $\mathbb{N}$. \qed
    \end{enumerate}
\end{exer}


\begin{exer}{3.2.2}{
    Use the axiom of regularity (and the singleton set axiom) to show that if $A$ is a set, then $A \notin A$.
    Furthermore, show that if $A$ and $B$ are two sets, then either $A \notin B$ or $B \notin A$ (or both).
}
    We will first prove that if $A$ is a set, then $A \notin A$.
    Consider the set $\{ A \}$ which exists by the singleton set axiom.
    By the axiom of regularity, there is at least one element $x \in \{ A \}$ such that $x$ is either not a set, or is disjoint from $\{ A \}$.
    From the singleton set axiom, we know that $x = A$.
    However, $A$ is a set, so $A$ and $\{ A \}$ must be disjoint sets, i.e. $A \cap \{ A \} = \varnothing$.
    If $A \in A$, then $A$ is an element of $A \cap \{ A \}$, which is contradictory.
    Therefore $A \notin A$ as desired.
    
    We will now show that if $A$ and $B$ are two sets, then either $A \notin B$ or $B \notin A$ (or both).
    Consider the set $\{ A, B \}$, which exists by the pair set axiom.
    According to the axiom of regularity, there is at least one element $x \in \{  A, B \}$ such that $x$ is either not a set, or is disjoint from $\{ A, B \}$.
    Since both $A$ and $B$ are sets, $x$ must be a set, thus $x$ must be disjoint from $\{ A, B \}$.
    If $x = A$, then $A$ is disjoint from $\{ A, B \}$.
    Thus, for any object $y$, $y \notin A \cap \{ A, B \}$, in particular, $B \notin A \cap \{ A, B \}$.
    Since $B \in \{ A, B \}$, it follows that $B \notin A$.
    Similarly, if $x = B$, we can deduce that $A \notin B$.
    Hence, either $B \notin A$ or $A \notin B$ as required. \qed
\end{exer}


\begin{exer}{3.2.3}{
    Show (assuming the other axioms of set theory) that the universal specification axiom, Axiom 3.8, is equivalent to an axiom postulating the existence of a ``universal set" $\Omega$ consisting of all objects (i.e., for all objects $x$, we have $x \in \Omega$).
    In other words, if Axiom 3.8 is true, then a universal set exists, and conversely, if a universal set exists, then Axiom 3.8 is true.
}
    We will first show the statement.
    Suppose Axiom 3.8 is true.
    Then, let $P(x)$ be a property pertaining to $x$ such that $x = x$.
    Thus, there exists a set $\{ x : x = x \}$ such that for every object $y$, $y \in \{ x : x = x\} \iff P(y)$.
    Because $P(y)$ is true for all $y$, all objects are elements of $\{ x : x = x \}$.
    Thus, the set $\{ x : x = x \}$ is a universal set.
    
    Now consider the converse.
    Suppose all objects are in a universal set $\Omega$.
    We want to show that there exists a set $\{ x : P(x) \}$ such that for every object $y$, $y \in \{x : P(x) \} \iff P(y)$.
    Let $A$ be a set, and let $P(x)$ be the property pertaining to $x$.
    By the axiom of specification, there exists a set $\{ x \in A : P(x) \}$ such that $y \in \{x \in A : P(x) \} \iff (y \in A \text{ and } P(y))$.
    Take $A = \Omega$.
    It follows that $x \in A$ for all $x$.
    Therefore, there exists a set $\{ x : P(x) \}$ such that $y \in \{x : P(x) \} \iff P(y)$, which is exactly the claim of Axiom 3.8. \qed
\end{exer}


\subsection{Functions}
\begin{exer}{3.3.1}{
    Show that the definition of equality in Definition 3.3.7 is reflexive, symmetric, and transitive.
    Also verify the substitution property: if $f, \tilde{f} : X \to Y$ and $g, \tilde{g} : Y \to Z$ are functions such that $f = \tilde{f}$ and $g = \tilde{g}$, then $g \circ f = \tilde{g} \circ \tilde{f}$.
}
    For the first section of this exercise, let $X$ and $Y$ be sets, and $f : X \to Y$, $g : X \to Y$, and $h : X \to Y$ be functions of the same domain and range.

    \begin{enumerate}[topsep=0mm, label=(\alph*)]
        \item \textit{Reflexivity: $f = f$.}
        
            Notice that $f$ and $f$ have the same domain and range.
            For every value $x \in X$, clearly $f(x) = f(x)$ by reflexivity of elements in set $Y$.
            Thus, by Definition 3.3.7, $f = f$.
        
        \item \textit{Symmetry: If $f = g$, $g = f$.}
            
            Suppose $f = g$.
            Then $f(x) = g(x)$ forall $x \in X$.
            But, from the symmetry of elements in set $Y$, for all $x \in X$, $g(x) = f(x)$.
            Heence $g = f$.
        
        \item \textit{Transitivity: If $f = g$ and $g = h$, then $f = h$.}
        
            Suppose $f = g$ and $g = h$.
            Then, $f(x) = g(x)$ and $g(x) = h(x)$.
            For any $x \in X$, $f(x) = g(x)$ and $g(x) = h(x)$.
            By the transitivity of elements in $Y$, we obtain $f(x) = h(x)$.
            Since $x$ is arbitrary, we can conclude that $f = h$.
           
    \end{enumerate}
    
    Suppose instead $X, Y, Z$ are sets, and suppose also $f, \tilde{f} : X \to Y$ and $g, \tilde{g} : Y \to Z$ such that $f = \tilde{f}$ and $g = \tilde{g}$.
    Thus, by Definition 3.3.7, $f(x) = \tilde{f} (x)$ for all $x \in X$ and $g(y) = \tilde{g} (y)$ for all $y \in Y$.
    By the substitution axiom, $g(f(x)) = g(\tilde{f} (x))$.
    Since $g = \tilde{g}$, $g(f(x)) = g(\tilde{f}(x)) = \tilde{g} (\tilde{f} (x))$, or $(g \circ f)(x) = (\tilde{g} \circ \tilde{f})(x).$
    By Definition 3.3.7, we obtain $g \circ f = \tilde{g} \circ \tilde{f}$ as required. \qed
\end{exer}


\begin{exer}{3.3.2}{
    Let $f : X \to Y$ and $g : Y \to Z$ be functions.
    Show that if $f$ and $g$ are both injective, then so is $g \circ f$;
    similarly, show that if $f$ and $g$ are both surjective, then so is $g \circ f$.
}
    Note that $g \circ f : X \to Z$.
    
    First, we will show that if $f$ and $g$ are both injective, then so is $g \circ f$.
    Suppose $g(f(x)) = g(f(x'))$, where $x, x' \in X$.
    Then, by injectivity of $g$, we obtain $f(x) = f(x')$.
    Similarly, by injectivity of $f$, we obtain $x = x'$.
    Therefore, $g \circ f$ is injective.

    We will now prove that if $f$ and $g$ are both surjective, then so is $g \circ f$.
    Because $f$ is surjective, for every $y \in Y$, there exists an $x \in X$ such that $f(x) = y$.
    Similarly, because $g$ is surjective, for every $z \in Z$, there exists an $y \in Y$ such that $g(y) = z$.
    Combining these two results, for every $z \in Z$, there exists an $x \in X$ such that $g(f(x)) = z$.
    Therefore, by definition, $g \circ f$ is surjective. \qed
\end{exer}


\begin{exer}{3.3.3}{
    When is the empty function injective? Surjective? Bijective?
}
    Let $X$ be a set, and $f : \varnothing \to X$ be the empty function.

    The function $f$ is considered \textbf{injective} if for any $p, p' \in \varnothing$ where $p \ne p'$, then $f(p) \ne f(p')$.
    This statement is vacuously true because there is no such $p$ and $p'$ in the empty set $\varnothing$.
    Thus, $f$ is always injective.

    The function $f$ is considered \textbf{surjective} if for any value $x$ in $X$, there is a $p \in \varnothing$ such that $f(p) = x$.
    This statement is false if $X \ne \varnothing$: As the empty set $\varnothing$ has no elements, no such $p$ exists.
    However, when $X = \varnothing$, this statement is vacuously true because no such $x$ exists.
    Thus, $f$ is surjective only when $X = \varnothing$.

    Since $f$ is always injective and surjective only when $X = \varnothing$, $f$ is only \textbf{bijective} when $X = \varnothing$. \qed
\end{exer}


\begin{exer}{3.3.4}{
    In this section we give some cancellation laws for composition.
    Let $f : X \to Y$, $\tilde{f} : X \to Y$, $g : Y \to Z$, and $\tilde{g} : Y \to Z$ be functions.
    Show that if $g \circ f = g \circ \tilde{f}$ and $g$ is injective, then $f = \tilde{f}$.
    Is the same statement true if $g$ is not injective?
    Show that if $g \circ f = \tilde{g} \circ f$ and $f$ is surjective, then $g = \tilde{g}$.
    Is the same statement true if $g$ is not surjective?
}
    We will first solve the first part of the exercise.
    Because $g$ is injective, if $g(f(x)) = g(\tilde{f}(x))$, then $f(x) = \tilde{f}(x)$ for any $x \in X$.
    Because $f$ and $\tilde{f}$ have the same domain and range, by Definition 3.3.7 (equality of functions), $f = \tilde{f}$.
    This statement is not true if $g$ is not injective.
    As a counterexample, take $g$ as a constant function, that is, $g(y) = c$ where $y \in Y$.
    Then, $g(f(x)) = g(\tilde{f}(x))$ for any $f$ and $\tilde{f}$, including when $f \ne \tilde{f}$.

    Now, consider the second part of the exercise.
    Because $f$ is surjective, for any $y \in Y$, there exists $x \in X$ such that $y = f(x)$.
    Then, $g(y) = \tilde{g}(y)$ for any $y \in Y$.
    Because $g$ and $\tilde{g}$ has the same domain and range, $g = \tilde{g}$.
    This statement is not true if $f$ is not surjective.
    As a counterexample, take $f$ as a constant function, that is, $f(x) = c$, where $x \in X$, $c \in Y$, and $Y \ne \{ c \}$.
    Evidently $f$ is not surjective.
    Thus, $g(c) = \tilde{g}(c)$ implies $g$ and $\tilde{g}$ are equal at $c$, but not necessarily at any other points.
    For instance, let $f(x) = 0$, $g(x) = x$, $\tilde{g}(x) = 2x$.
    Then, $g \ne \tilde{g}$ even though $g(0) = \tilde{g}(0)$. \qed    
\end{exer}

\begin{exer}{3.3.5}{
    Let $f : X \to Y$ and $g : Y \to Z$ be functions.
    Show that if $g \circ f$ is injective, then $f$ must be injective.
    Is it true that $g$ must also be injective?
    Show that if $g \circ f$ is surjective, then $g$ must be surjective.
    Is it true that $f$ must also be surjective?
}
    We will first solve the first part of the exercise.
    For the sake of contradiction, suppose $f$ isn't injective.
    In other words, suppose there exist objects $x, x' \in X$ such that $x \ne x'$ and $f(x) = f(x')$.
    This leads to $g(f(x)) = g(f(x'))$.
    But $x \ne x'$, which implies $g \circ f$ is not injective, which is contradictory.
    Thus, \textbf{$f$ must be injective}.
    However, $g$ need not be injective.
    As a counterexample, let $X = \{ 0, 1 \}$, $Y = Z = \{ 0, 1, 2 \}$ and $f(0) = 0$, $f(1) = 1$.
    Let $g$ such that $g(0) = 0, g(1) = 1, g(2) = 0$ such that $g \circ f$ is injective.
    Evidently, $g$ is not injective.

    Now consider the second part of the exercise.
    For the sake of contradiction, suppose $g$ is not surjective.
    In other words, there exists a $z \in Z$ such that for all $y \in Y$, $g(y) \ne z$.
    In particular, since $f(x) \in Y$, for all $x \in X$, $g(f(x)) \ne z$.
    This contradicts the condition that $g \circ f$ is surjective.
    Hence, \textbf{$g$ must be surjective}.
    However, $f$ need not be surjective.
    As a counterexample, let $X = Y = \{ 0, 1 \}$, $Z = \{ 0 \}$.
    We define $f$ such that $f(0) = 0$, $f(1) = 0$, and $g$ such that $g(0) = 0$, $g(1) = 0$.
    In this case, $g \circ f$ is surjective, but $f$ is not. \qed
\end{exer}


\begin{exer}{3.3.6}{
    Let $f: X \to Y$ be a bijective function, and let $f^{-1} : Y \to X$ be its inverse.
    Verify the cancellation laws $f^{-1} (f(x)) = x$ for all $x \in X$ and $f(f^{-1}(y)) = y$ for all $y \in Y$.
    Conclude that $f^{-1}$ is also invertible, and has $f$ has its inverse (thus $(f^{-1})^{-1} = f$).
}
    Suppose $x$ is an arbitrary element of $X$, and $y = f(x)$, $y \in Y$.
    By the definition of inverse functions, because $f$ is bijective, for every $y \in Y$, there is exactly one $x' \in X$ such that $y = f(x')$, which is $x' = f^{-1}(y)$.
    Because $f(x) = y = f(x')$, the injectivity of $f$ leads to $x = x'$.
    Thus, $x = f^{-1}(y) = f^{-1}(f(x))$ as required.

    Suppose instead $y$ is an arbitrary element of $Y$.
    Since $f$ is bijective, there is exactly one value of $x \in X$ such that $f(x) = y$.
    This value of $x$ is denoted as $f^{-1}(y)$.
    Thus, $f(f^{-1}(y)) = y$ as desired.

    Now, to show that $f^{-1}$ is also invertible, we need to show that $f^{-1}$ is bijective.
    In other words, we need to prove the injectivity and surjectivity of $f^{-1}$.
    Suppose $f^{-1}(y) = f^{-1}(y')$ for $y, y' \in Y$.
    Applying $f$ to both sides, we obtain $f(f^{-1}(y)) = f(f^{-1}(y'))$, which leads to $y = y'$ from previous results.
    Thus, $f^{-1}$ is injective.
    Because $f$ is bijective, for every $x \in X$, there exists $y \in Y$ such that $f(x) = y$.
    This is precisely the definition for $f^{-1}$ to be surjective.
    Hence, we have shown that $f^{-1}$ is bijective as required. \qed
\end{exer}


\begin{exer}{3.3.7}{
    Let $f : X \to Y$ and $g : Y \to Z$ be functions.
    Show that if $f$ and $g$ are bijective, then so is $g \to f$, and we have $(g \circ f)^{-1} = f^{-1} \circ g^{-1}$.
}
    From Exercise 3.3.2, we know $g \circ f$ is both injective and surjective; thus $g \circ f$ is bijective.
    In this exercise, we only need to show $(g \circ f)^{-1} = f^{-1} \circ g^{-1}$.

    Let $z$ be an arbitrary element of $Z$.
    Because $g$ is bijective, for every $z \in Z$, there exists exactly one $y \in Y$ such that $g(y) = z$, or $y = g^{-1}(z)$.
    Similarly, because $f$ is bijective, for every $y \in Y$, there exists exactly one $x \in X$ such that $f(x) = y$, or $x = f^{-1}(y)$.
    Thus, for every $z \in Z$, there exists exactly one $x \in X$ such that $(g \circ f)(x) = z$, $x = f^{-1}(g^{-1}(z))$.
    Therefore, $x = (g \circ f)^{-1}(z) = (f^{-1} \circ g^{-1})(z)$, or $(g \circ f)^{-1} = f^{-1} \circ g^{-1}$ as desired. \qed
\end{exer}


\begin{exer}{3.3.8}{
    If $X$ is a subset of $Y$, let $\iota_{X \to Y} : X \to Y$ be the \emph{inclusion map from $X$ to $Y$} defined by mapping $x \mapsto x$for all $x \in X$, i.e. $\iota_{X \to Y}(x) \coloneqq x$ for all $x \in X$.
    The map $\iota_{X \to X}$ in particular is called the \emph{identity map} on $X$.
}
    \begin{enumerate}[topsep=0mm, label=(\alph*)]
        \item \textit{Show that if $X \subseteq Y \subseteq Z$ then $\iota_{Y \to Z} \circ \iota_{X \to Y} = \iota_{X \to Z}$.}

            Note that $\iota_{Y \to Z} \circ \iota_{X \to Y}$ and $\iota_{X \to Z}$ have the same domain and range, and that the range of $\iota_{X \to Y}$ is equal to the domain of $\iota_{Y \to Z}$.
            Let $x$ be an arbitrary element of $X$.
            Thus, $\iota_{Y \to Z}(\iota_{X \to Y}(x)) = \iota_{Y \to Z}(x) = x$ because $x \in Y$.
            Also, $\iota_{X \to Z}(x) = x$.
            Hence, $\iota_{Y \to Z}(\iota_{X \to Y}(x)) = \iota_{X \to Z}(x)$ for all $x \in X$, thus $\iota_{Y \to Z} \circ \iota_{X \to Y} = \iota_{X \to Z}$.

        \item \textit{Show that if $f : A \to B$ is any function, then $f = f \circ \iota_{A \to A} = \iota_{B \to B} \circ f$.}

            The compositions $f \circ \iota_{A \to A}$ and $\iota_{B \to B} \circ f$ have the same domain and range as $f$, so we just have to prove these functions agree on all inputs.
            Let $a$ be an arbitrary element of $A$.
            Then, $f(\iota_{A \to A}(a)) = f(a)$.
            Also, $\iota_{B \to B}(f(a)) = f(a)$.
            As all three functions have the same output for all $a \in A$, $f = f \circ \iota_{A \to A} = \iota_{B \to B} \circ f$.

        \item \textit{Show that, if $f : A \to B$ is a bijective function, then $f \circ f^{-1} = \iota_{B \to B}$ and $f^{-1} \circ f = \iota_{A \to A}$.}

            Let $b$ be an arbitrary element of $B$.
            Since $f$ is bijective, there exists exactly one $a \in A$ such that $f(a) = b$.
            Thus, $f(f^{-1}(b)) = f(a) = b = \iota_{B \to B}(b)$ for all $b \in B$.
            Because $f \circ f^{-1}$ and $\iota_{B \to B}$ have the same domain and range, $f \circ f^{-1} = \iota_{B \to B}$.
            In a similar fashion, we can show that $f^{-1} \circ f = \iota_{A \to A}$ as required.

        \item \textit{Show that if $X$ and $Y$ are disjoint sets, and $f : X \to Z$ and $g : Y \to Z$ are functions, then there is a unique function $h : X \cup Y \to Z$ such that $h \circ \iota_{X \to X \cup Y} = f$ and $h \circ \iota_{Y \to X \cup Y} = g$.}

            We will first show the existence of the function $h$.
            Define $h$ as such: for any $p \in X \cup Y$, if $p \in X$, then $h(p) \coloneqq f(p)$; if $p \in Y$, then $h(p) \coloneqq g(p)$.
            Because $p$ must be in exactly one of $X$ and $Y$, this definition makes sense.
            Thus, $h(\iota_{X \to X \cup Y}(x)) = h(x) = f(x)$ for any $x \in X$ and $h(\iota_{Y \to X \cup Y}(y)) = g(y)$ for any $y \in Y$.
            This matches the given definition of $h$, hence the function $h$ exists.

            We will now show the uniqueness of the function $h$.
            Suppose $\tilde{h} : X \cup Y \to Z$ is another function such that $\tilde{h} \circ \iota_{X \to X \cup Y} = f$ and $\tilde{h} \circ \iota_{Y \to X \cup Y} = g$.
            Let $p$ be an element in $X \cup Y$.
            If $p \in X$, then $\iota_{X \to X \cup Y}(p) = p$, thus $\tilde{h}(p) = \tilde{h}(\iota_{X \to X \cup Y}(p)) = f(p)$. However, this is equal to $h(p)$.
            Similarly, if $p \in Y$, $\tilde{h}(p) = h(p)$.
            Because $\tilde{h}(p) = h(p)$ for all $p \in X \cup Y$, $\tilde{h} = h$.
            Hence, the function $h$ is unique. \qed
    \end{enumerate}
\end{exer}



\subsection{Images and inverse images}
\begin{exer}{3.4.1}{
    Let $f : X \to Y$ be a bijective function, and let $f^{-1} : Y \to X$ be its inverse.
    Let $V$ be any subset of $Y$.
    Prove the forward image of $V$ under $f^{-1}$ is the same set as the inverse image of $V$ under $f$; thus, the fact that both sets are denoted by $f^{-1}(V)$ will not lead to any inconsistencies.
}
    Let $F$ be the forward image of $V$ under $f^{-1}$, and $I$ be the inverse image of $V$ under $f$.
    Thus, by Definition 3.4.1 and Definition 3.4.4, $F = \{f^{-1}(x) : x \in V \}$ and $I = \{x \in X : f(x) \in V \}$ respectively.
    We need to show $F = I$, or $F \subseteq I$ and $I \subseteq F$.

    We will first show $F \subseteq I$.
    Let $a$ be an arbitrary element of $F$.
    Then, there exists a $b \in V$ such that $a = f^{-1}(b)$, or $b = f(a)$, thus $f(a) \in V$.
    Also, because $F \subseteq X$ and $a = f^{-1}(b)$, $a \in X$.
    Thus, we have both $a \in X$ and $f(a) \in V$, which implies $a \in I$, hence $F \subseteq I$.

    Consider instead the statement $I \subseteq F$.
    Suppose $c \in I$, then $f(c) \in V$.
    Write $d = f(c)$.
    Thus, we have $c \in I$ and $c = f^{-1}(d)$ for some $d \in V$.
    This implies $c \in F$ by definition of forward images, hence $I \subseteq F$.
\end{exer}


\begin{exer}{3.4.2}{
    Let $f : X \to Y$ be a function from one set $X$ to another set $Y$, let $S$ be a subset of $X$, and let $U$ be a subset of $Y$.
    What, in general, can one say about $f^{-1}(f(S))$ and $S$?
    What about $f(f^{-1}(U))$ and $U$?
}
    We will consider the containment of each pair of sets.
    Generally, we claim $S \ne f^{-1}(f(S))$, but $S \subseteq f^{-1}(f(S))$; and $f(f^{-1}(U)) \ne U$, but $f(f^{-1}(U)) \subseteq U$.
    
    First consider $f^{-1}(f(S))$ and $S$.
    To show $f^{-1}(f(S)) \subseteq S$ is generally false, we will consider a counterexample.
    Let $f : \{ 0, 1 \} \to \{ 1 \}$ such that $f(0) = f(1) = 1$, and $S = \{ 0 \}$.
    Thus, $f(S) = \{ 1 \}$ and $f^{-1}(f(S)) = \{ 0, 1 \}$, which is not a subset of $S$.
    Now, to prove $S \subseteq f^{-1}(f(S))$, suppose $x \in S$, thus $x \in X$.
    Then, by Definition 3.4.1, $f(x) \in f(S)$.
    This leads to $x \in f^{-1}(f(S))$ by Definition 3.4.4 because $x \in X$ and $f(x) \in f(S)$, which implies $S \subseteq f^{-1}(f(S))$ as desired.

    Now consider $f(f^{-1}(U))$ and $U$.
    To show $U \subseteq f(f^{-1}(U))$ is generally false, we will consider a counterexample.
    Let $f : \{0 \} \to \{ 0, 1 \}$ such that $f(0) = 1$, and $U = \{ 0, 1 \}$.
    Thus, $f^{-1}(U) = \{ 0 \}$ and $f(f^{-1}(U)) = \{ 1 \}$, which $U$ is not a subset of.
    However, to show $f(f^{-1}(U)) \subseteq U$, suppose $x \in f(f^{-1}(U))$.
    Then, there exists some $y \in f^{-1}(U)$ such that $x = f(y)$.
    According to Definition 3.4.4, $y \in f^{-1}(U)$ if and only if $f(y) \in U$, thus $x \in U$ because $x = f(y)$.
    Because $x \in U$ for all $x \in f(f^{-1}(U))$, we obtain $f(f^{-1}(U)) \subseteq U$ as desired. \qed
\end{exer}


\begin{exer}{3.4.3}{
    Let $A, B$ be two subsets of a set $X$, and let $f : X \to Y$ be a function.
    Show that $f(A \cap B) \subseteq f(A) \cap f(B)$, that $f(A) \setminus f(B) \subseteq f(A \setminus B)$, $f(A \cup B) = f(A) \cup f(B)$.
    For the first two statements, is it true that the $\subseteq$ relation can be improved to $=$?
}
    \begin{enumerate}[topsep=0mm, label=(\alph*)]
        \item $f(A \cap B) \subseteq f(A) \cap f(B)$.

            Let $y \in f(A \cap B)$.
            Then, there exists some $x \in A \cap B$ such that $y = f(x)$.
            This implies $x \in A$ and $x \in B$.
            Because $y = f(x)$, $y \in f(A)$ and $y \in f(B)$, which leads to $y \in f(A) \cap f(B)$.
            Because $y \in f(A) \cap f(B)$ for all $y \in f(A \cap B)$, $f(A \cap B) \subseteq f(A) \cap f(B)$.

            Consider the reverse inclusion instead.
            As a counterexample, let $A = \{ 0 \}$, $B = \{ 1 \}$, $f(x) = 5$.
            Thus, $A \cap B = \varnothing$, and $f(\varnothing) = \varnothing$.
            However, $f(A) = \{ 5 \}$, $f(B) = \{ 5 \}$, and $f(A) \cap f(B) = \{ 5 \}$, which is not a subset of $f(A \cap B) = \varnothing$.
            Hence, the reverse inclusion is false.

        \item $f(A) \setminus f(B) \subseteq f(A \setminus B)$.

            Let $y \in f(A) \setminus f(B)$; thus $y \in f(A)$ and $y \notin f(B)$.
            Then, there exists an $x \in A$ such that $y = f(x)$.
            If $x \in B$, then $y \in f(B)$, therefore $x \notin B$.
            Thus, we have $y = f(x)$ for some $x \in A \setminus B$, or $y \in f(A \setminus B)$ for all $y \in f(A) \setminus f(B)$.
            Hence, $f(A) \setminus f(B) \subseteq f(A \setminus B)$.

            Consider the reverse inclusion instead.
            As a counterexample, let $A = \{ 1, 2 \}$, $B = \{ 1 \}$, and $f(x) = 4$ for any $x$.
            Thus, $f(A) = f(B) = \{ 4 \}$, and $f(A) \setminus f(B) = \varnothing$.
            However, $A \setminus B = \{ 2 \}$, and $f(A \setminus B) = \{ 4 \}$, which is not a subset of $f(A) \setminus f(B) = \varnothing$.
            Hence, the reverse inclusion is false.

        \item $f(A \cup B) = f(A) \cup f(B)$.

            We will first consider $f(A \cup B) \subseteq f(A) \cup f(B)$.
            Let $y \in f(A \cup B)$.
            Then, there exists an $x \in A \cup B$ such that $y = f(x)$.
            Thus, $x \in A$ or $x \in B$.
            If $x \in A$, then $y = f(x) \in f(A)$, which implies $y \in f(A) \cup f(B)$.
            Similarly, if $x \in B$, $y \in f(A) \cup f(B)$.
            Because $y \in f(A) \cup f(B)$ for all $y \in f(A \cup B)$, we can conclude that $f(A \cup B) \subseteq f(A) \cup f(B)$.

            Now consider the reverse inclusion.
            Let $y \in f(A) \cup f(B)$.
            Then, $y \in f(A)$ or $y \in f(B)$.
            If $y \in f(A)$, then there exists an $x \in A$ such that $y = f(x)$.
            Therefore, $x \in A \cup B$, and it follows that $y \in f(A \cup B)$.
            Similarly, if $y \in f(B)$, then $y \in f(A \cup B)$.
            Because $y \in f(A \cup B)$ for all $y \in f(A) \cup f(B)$, we can conclude that $f(A) \cup f(B) \subseteq f(A \cup B)$.
            Hence, $f(A \cup B) = f(A) \cup f(B)$. \qed
    \end{enumerate}
\end{exer}


\begin{exer}{3.4.4}{
    Let $f : X \to Y$ be a function from one set $X$ to another set $Y$, and let $U$, $V$ be subsets of $Y$.
    Show that $f^{-1}(U \cup V) = f^{-1}(U) \cup f^{-1}(V)$,
    that $f^{-1}(U \cap V) = f^{-1}(U) \cap f^{-1}(V)$,
    and that $f^{-1}(U \setminus V) = f^{-1}(U) \setminus f^{-1}(V)$.
}
    \begin{enumerate}[topsep=0mm, label=(\alph*)]
        \item $f^{-1}(U \cup V) = f^{-1}(U) \cup f^{-1}(V)$.
        
            Suppose $x \in f^{-1}(U \cup V)$.
            By definition, $f(x) \in U \cup V$; it follows that $f(x) \in U$ or $f(x) \in V$.
            If $f(x) \in U$, then $x \in f^{-1}(U)$, and $x \in f^{-1}(U) \cup f^{-1}(V)$.
            Similarly, if $f(x) \in V$, then $x \in f^{-1}(U) \cup f^{-1}(V)$.
            Therefore, for any $x \in f^{-1}(U \cup V)$, we have $x \in f^{-1}(U) \cup f^{-1}(V)$.
            
            Now consider the converse.
            Suppose $x \in f^{-1}(U) \cup f^{-1}(V)$, thus $x \in f^{-1}(U)$ or $x \in f^{-1}(V)$.
            If $x \in f^{-1}(U)$, then $f(x) \in U$, and $f(x) \in U \cup V$.
            Similarly, if $x \in f^{-1}(V)$, then $f(x) \in U \cup V$.
            This leads to $x \in f^{-1}(U \cup V)$ for all $x \in f^{-1}(U) \cup f^{-1}(V)$.
            Hence, $f^{-1}(U \cup V) = f^{-1}(U) \cup f^{-1}(V)$.         
        
        \item $f^{-1}(U \cap V) = f^{-1}(U) \cap f^{-1}(V)$.

            Suppose $x \in f^{-1}(U \cap V)$.
            By definition, $f(x) \in U \cap V$; it follows that $f(x) \in U$ and $f(x) \in V$.
            Therefore, $x \in f^{-1}(U)$ and $x \in f^{-1}(V)$, thus $x \in f^{-1}(U) \cap f^{-1}(V)$.
            
            Now consider the converse.
            Suppose $x \in f^{-1}(U) \cap f^{-1}(V)$, thus $x \in f^{-1}(U)$ and $x \in f^{-1}(V)$.
            By definition, $f(x) \in U$ and $f(X) \in V$, thus $f(x) \in U \cap V$.
            This leads to $x \in f^{-1}(U \cap V)$ for all $x \in f^{-1}(U) \cap f^{-1}(V)$.
            Hence, $f^{-1}(U \cap V) = f^{-1}(U) \cap f^{-1}(V)$.            
        
        \item $f^{-1}(U \setminus V) = f^{-1}(U) \setminus f^{-1}(V)$.

            Suppose $x \in f^{-1}(U \setminus V)$.
            By definition, $f(x) \in U \setminus V$; it follows that $f(x) \in U$ and $f(x) \notin V$.
            Therefore, $x \in f^{-1}(U)$ and $x \notin f^{-1}(V)$, thus $x \in f^{-1}(U) \setminus f^{-1}(V)$.
            
            Now consider the converse.
            Suppose $x \in f^{-1}(U) \setminus f^{-1}(V)$, thus $x \in f^{-1}(U)$ and $x \notin f^{-1}(V)$.
            By definition, $f(x) \in U$ and $f(x) \notin V$, thus $f(x) \in U \setminus V$.
            This leads to $x \in f^{-1}(U \setminus V)$ for all $x \in f^{-1}(U) \setminus f^{-1}(V)$.
            Hence, $f^{-1}(U \setminus V) = f^{-1}(U) \setminus f^{-1}(V)$. \qed    
    \end{enumerate}
\end{exer}


\begin{exer}{3.4.5}{
    Let $f : X \to Y$ be a function from one set $X$ to another set $Y$.
    Show that $f(f^{-1}(S)) = S$ for every $S \subseteq Y$ if and only if $f$ is surjective.
    Show that $f^{-1}(f(S))$ for every $S \subseteq X$ if and only if $f$ is injective.
}
    \begin{enumerate}[topsep=0mm, label=(\alph*)]
        \item Show that $f(f^{-1}(S)) = S$ for every $S \subseteq Y$ if and only if $f$ is surjective.

            We will first prove the implication.
            Suppose $f(f^{-1}(S)) = S$ for every $S \subseteq Y$; we need to show that for all $y \in Y$, there exists an $x \in X$ such that $f(x) = y$.
            Consider the subset $S = Y$, which contains all elements of $Y$.
            By the initial assumption, we have $Y = f(f^{-1}(Y))$, thus $y \in f(f^{-1}(Y))$ for any $y \in Y$.
            From the definition of forward images, there exists an $x \in f^{-1}(Y)$ such that $f(x) = y$, which is exactly our claim.

            Now consider the converse.
            Suppose $f$ is surjective; we need to show that $f(f^{-1}(S)) = S$.
            Let $y$ be an arbitrary element of $S$.
            Because $f$ is surjective, there exists an $x \in X$ such that $f(x) = y$.
            Thus, $f(x) \in S$, which implies $x \in f^{-1}(S)$ by the definition of inverse images.
            From the definition of forward images, $f(x) \in f(f^{-1}(S))$, which leads to $y \in f(f^{-1}(S))$.
            From Exercise 3.4.2, we know $f(f^{-1}(S)) \subseteq S$.
            Hence, $f(f^{-1}(S)) = S$ if $f$ is surjective.

            As the implication and its converse are both true, we have $f(f^{-1}(S)) = S$ for every $S \subseteq Y$ if and only if $f$ is surjective.         
        
        \item Show that $f^{-1}(f(S)) = S$ for every $S \subseteq X$ if and only if $f$ is injective.

            We will first prove the implication.
            Suppose $f^{-1}(f(S)) = S$ for every $S \subseteq X$; we need to show that $f$ is injective.
            Let $x, y \in X$ such that $f(x) = f(y)$.
            We want to show that $x = y$.
            Take $S = \{ x \}$.
            Then, $f(\{ x \}) = \{ f(x) \}$.
            Since $f(x) = f(y)$, $f(y)$ is an element of $f(\{ x \})$, therefore $y \in f^{-1}(f(\{ x \}))$ by the definition of inverse images.
            Also, by the hypothesis, we have $f^{-1}(f(\{ x \})) = \{ x \}$.
            Therefore, we have $y \in \{ x \}$, which implies $y = x$.
            Hence, we can conclude that if $f^{-1}(f(S)) = S$ for every $S \subseteq X$, then $f$ is injective.

            Now consider the converse.
            Suppose $f$ is injective; we need to show $f^{-1}(f(S)) = S$ for every $S \subseteq X$.
            Let $x \in f^{-1}(f(S))$.
            Then, by definition, we have $f(x) \in f(S)$, and thus there exists some $x' \in S$ such that $f(x) = f(x')$.
            Because $f$ is injective, we have $x = x'$, thus $x \in S$.
            Thus, $f^{-1}(f(S)) \subseteq S$.
            From Exercise 3.4.2, we know $S \subseteq f^{-1}(f(S))$.
            Hence, we can conclude that if $f$ is injective, then $f^{-1}(f(S)) = S$ for every $S \subseteq X$.

            Because the implication and its converse are both true, we have $f^{-1}(f(S)) = S$ for every $S \subseteq X$ if and only if $f$ is injective. \qed
    \end{enumerate}
\end{exer}


\begin{exer}{3.4.6}{
    Prove Lemma 3.4.9:
    Let $X$ be a set.
    Then the set
    \[
        \{ Y : Y \text{ is a subset of } X \}%
    \] is a set.
}
    By the power set axiom, $\{ 0, 1 \}^X$ is a set, and its elements are exactly the functions mapping from $X$ to $\{ 0, 1 \}$.
    For each function $f \in \{ 0, 1 \}^X$, define $A_f \coloneqq f^{-1}(\{1\})$.
    Since $f$ has a domain of $X$, every element of $A_f$ is an element of $X$; thus we have $A_f \subseteq X$.
    
    Let $A$ be an arbitrary subset of $X$.
    Then, we can define the function $f_A : X \to \{ 0, 1 \}$ such that for all $x \in X$, we have $f_A(x) = 1$ if $x \in A$, and $f_A(x) = 0$ otherwise.
    Note that $f^{-1}_A(\{  1 \}) = A$.
    Also, because $f_A$ has domain $X$ and range $\{ 0, 1 \}$, we have $f_A \in \{ 0, 1 \}^X$.
    Thus, $A$ is a subset of $X$ if and only if $A = f^{-1}(X)$ for some $f \in \{ 0, 1 \}^X$.
    For each $f$, there is exactly one $A$ such that $f^{-1}(\{1\}) =A$.
    Hence, by the axiom of replacement, the collection $\{ A : f^{-1}(\{ 1 \}) \text{ for some } f \in \{ 0, 1 \}^X \}$ exists.
    This is equivalent to saying $\{ A : A \text{ is a subset of } X \}$. \qed 
\end{exer}


\begin{exer}{3.4.7}{
    Let $X$, $Y$ be sets. Define a \emph{partial function} from $X$ to $Y$ to be any function $f : X' \to Y'$ whose domain $X'$ is a subset of $X$, and whose range $Y'$ is a subset of $Y$. 
    Show that the collection of all partial functions from $X$ to $Y$ is itself a set.
}
    From the problem statement, we have $X' \subseteq X$ and $Y' \subseteq Y$.
    For any $X'$ and $Y'$, we have $Y'^{X'}$, a set with all functions mapping from $X'$ to $Y'$.
    Also, by Lemma 3.4.9, we have the set $2^X$, the set of all subsets of $X$, and $2^Y$, the set of all subsets of $Y$.
    Thus, $X' \in 2^X$ and $Y' \in 2^Y$.

    We first fix a certain $X'$.
    Let $A$ be a set, and $P(Y', A)$ be the property pertaining to $Y'$ that $A = Y'^{X'}$.
    Then, for any $Y' \in 2^Y$, $P(Y', A)$ will be true for exactly one $A$.
    Therefore, by the replacement axiom, the set $\{ A : A = Y'^{X'} \text{ for some } Y' \in 2^Y \} = \{ Y'^{X'} : Y' \in 2^Y \}$ exists.
    Because each of the elements of this set is a set, we also have the set $\bigcup \{ Y'^{X'} : Y' \in 2^Y \} = \{ f \mid f : X' \to Y' \text{ for some } Y' \in 2^Y \}$.
    Write this set as $S_{X'}$.

    Now consider $X' \in 2^X$.
    For every $X'$, we have exactly one set $S_{X'}$.
    According to the replacement axiom (Axiom 3.6), the set $\{S_{X'} : X' \in 2^X \}$ exists.
    By the union axiom (Axiom 3.11), this means the set $\bigcup_{X' \in 2^X} S_{X'}$ exists.
    For any function $f$, $f$ is an element of this set if and only if we have $f : X' \to Y'$ for some $X' \subseteq X$ and $Y' \subseteq Y$.
    This is precisely the collection of all partial functions from $X$ to $Y$. \qed
\end{exer}


\begin{exer}{3.4.8}{Show that Axiom 3.4 can be deduced from Axiom 3.1, Axiom 3.3 and Axiom 3.11.}
    Let $A$ and $B$ be sets.
    According to Axiom 3.1, the sets $A$ and $B$ are objects, and therefore can be elements of sets.
    Therefore, we can construct the pair set $\{ A, B \}$.
    From Axiom 3.11, there exists a set $\bigcup \{ A, B \}$ such that for any $x \in \bigcup \{ A, B \}$, $x \in S$ for some $S \in \{ A, B \}$.
    However, since $\{ A, B \}$ is a pair set with two elements, we have $S = A$ or $S = B$.
    Hence, $x \in A$ or $x \in B$ if and only if $x \in \bigcup \{ A, B \}$.
    We denote $\bigcup\{ A, B \} = A \cup B$; this is precisely the definition of pairwise union as in Axiom 3.4. \qed
\end{exer}


\begin{exer}{3.4.9}{Show that if $\beta$ and $\beta'$ are two elements of a set $I$, and to each $\alpha \in I$ we assign a set $A_\alpha$, then

\begin{exer}{3.4.10}{
    Suppose that $I$ and $J$ are two sets, and for all $\alpha \in I \cup J$ let $A_\alpha$ be a set.
    Show that $(\bigcup_{\alpha \in I} A_\alpha) \cup (\bigcup_{\alpha \in J} A_\alpha) = \bigcup_{\alpha \in I \cup J} A_\alpha$.
    If $I$ and $J$ are non-empty, show that $(\bigcap_{\alpha \in I} A_\alpha) \cap (\bigcap_{\alpha \in J} A_\alpha) = \bigcap_{\alpha \in I \cup J} A_\alpha$.
}
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item Show that $(\bigcup_{\alpha \in I} A_\alpha) \cup (\bigcup_{\alpha \in J} A_\alpha) = \bigcup_{\alpha \in I \cup J} A_\alpha$.
    
            We will first consider the left-to-right implication.
            Let $y \in (\bigcup_{\alpha \in I} A_\alpha) \cup (\bigcup_{\alpha \in J} A_\alpha)$.
            Then, $y \in \bigcup_{\alpha \in I} A_\alpha$ or $y \in \bigcup_{\alpha \in J} A_\alpha$.
            If $y \in \bigcup_{\alpha \in I} A_\alpha$, then $y \in A_\alpha$ for some $\alpha \in I$.
            This implies $y \in A_\alpha$ for some $\alpha \in I \cup J$, or $y \in \bigcup_{\alpha \in I \cup J} A_\alpha$.
            Using a similar argument, if $y \in \bigcup_{\alpha \in J} A_\alpha$, then we have $y \in \bigcup_{\alpha \in I \cup J} A_\alpha$.
            Thus, we have $(\bigcup_{\alpha \in I} A_\alpha) \cup (\bigcup_{\alpha \in J} A_\alpha) \subseteq \bigcup_{\alpha \in I \cup J} A_\alpha$.
        
            Now consider the converse.
            Let $y \in \bigcup_{\alpha \in I \cup J} A_\alpha$.
            Then, $y \in A_\alpha$ for some $\alpha \in I \cup J$.
            We will split into two cases.
            If $y \in A_\alpha$ for some $\alpha \in I$, then $y \in \bigcup_{\alpha \in I} A_\alpha$.
            Similarly, if $y \in A_\alpha$ for some $\alpha \in J$, then $y \in \bigcup_{\alpha \in J} A_\alpha$.
            Therefore, we have $y \in (\bigcup_{\alpha \in I} A_\alpha) \cup (\bigcup_{\alpha \in J} A_\alpha)$.
            This is the same as $\bigcup_{\alpha \in I \cup J} A_\alpha \subseteq (\bigcup_{\alpha \in I} A_\alpha) \cup (\bigcup_{\alpha \in J} A_\alpha)$.
        
            Therefore, we have $(\bigcup_{\alpha \in I} A_\alpha) \cup (\bigcup_{\alpha \in J} A_\alpha) = \bigcup_{\alpha \in I \cup J} A_\alpha$.

        \item If $I$ and $J$ are non-empty, show that $(\bigcap_{\alpha \in I} A_\alpha) \cap (\bigcap_{\alpha \in J} A_\alpha) = \bigcap_{\alpha \in I \cup J} A_\alpha$.

            We will first consider the left-to-right implication.
            Let $y \in (\bigcap_{\alpha \in I} A_\alpha) \cap (\bigcap_{\alpha \in J} A_\alpha)$.
            Then, $y \in \bigcap_{\alpha \in I} A_\alpha$ and $y \in \bigcap_{\alpha \in J} A_\alpha$.
            This means $y \in A_\alpha$ for all $\alpha \in I$ and for all $\alpha \in J$.
            Thus, we have $y \in A_\alpha$ for all $\alpha \in I \cup J$, or $y \in \bigcap_{\alpha \in I \cup J} A_\alpha$.
            Hence, $(\bigcap_{\alpha \in I} A_\alpha) \cap (\bigcap_{\alpha \in J} A_\alpha) \subseteq \bigcap_{\alpha \in I \cup J} A_\alpha$.

            Now consider the converse.
            Let $y \in \bigcap_{\alpha \in I \cup J} A_\alpha$.
            Then, $y \in A_\alpha$ for all $\alpha \in (I \cup J)$.
            This leads to $y \in A_\alpha$ for all $\alpha \in I$ and for all $\alpha \in J$.
            Therefore, we have $y \in \bigcap_{\alpha \in I} A_\alpha$ and $y \in \bigcap_{\alpha \in J} A_\alpha$.
            We then have $y \in (\bigcap_{\alpha \in I} A_\alpha) \cap (\bigcap_{\alpha \in J} A_\alpha)$.
            Hence, $\bigcap_{\alpha \in I \cup J} A_\alpha \subseteq (\bigcap_{\alpha \in I} A_\alpha) \cap (\bigcap_{\alpha \in J} A_\alpha)$.

            Therefore, we have $(\bigcap_{\alpha \in I} A_\alpha) \cap (\bigcap_{\alpha \in J} A_\alpha) = \bigcap_{\alpha \in I \cup J} A_\alpha$ as required. \qed
    \end{enumerate}
\end{exer}


\begin{exer}{3.4.11}{
    Let $X$ be a set, let $I$ be a non-empty set, and for all $\alpha \in I$ let $A_\alpha$ be a subset of $X$.
    Show that
    \[
        X \setminus\bigcup_{\alpha \in I} A_\alpha = \bigcap_{\alpha \in I}(X \setminus A_\alpha)
    \text{ and }
        X \setminus\bigcap_{\alpha \in I} A_\alpha = \bigcup_{\alpha \in I}(X \setminus A_\alpha).
    \]
}
    We will prove the first equation.
    Let $x \in X \setminus \bigcup_{\alpha \in I} A_\alpha$.
    Then, we have
    \begin{align*}
             x \in X \setminus \bigcup_{\alpha \in I} A_\alpha
        \iff & x \in X \text{ and } x \notin \bigcup_{\alpha \in I} A_\alpha \\
        \iff & x \in X \text{ and } x \notin A_\alpha \text{ for any } \alpha \in I  \\
        \iff & x \in X \setminus A_\alpha \text{ for any } \alpha \in I  \\
        \iff & x \in \bigcap_{\alpha \in I} (X \setminus A_\alpha)
    \end{align*}
    Thus, $X \setminus\bigcup_{\alpha \in I} A_\alpha = \bigcap_{\alpha \in I}(X \setminus A_\alpha)$.

    We will now prove the second equation in a similar manner.
    Let $x \in X \setminus \bigcap_{\alpha \in I} A_\alpha$.
    Then, we have
    \begin{align*}
             x \in X \setminus \bigcap_{\alpha \in I} A_\alpha
        \iff & x \in X \text{ and } x \notin \bigcap_{\alpha \in I} A_\alpha  \\
        \iff & x \in X \text{ and } x \notin A_\alpha \text{ for some } \alpha \in I  \\
        \iff & x \in X \setminus A_\alpha \text{ for some } \alpha \in I  \\
        \iff & x \in \bigcup_{\alpha \in I} (X \setminus A_\alpha)
    \end{align*}
    Thus, $X \setminus\bigcap_{\alpha \in I} A_\alpha = \bigcup_{\alpha \in I}(X \setminus A_\alpha)$ as required. \qed

\subsection{Cartesian products}
\begin{exer}{3.5.1}{
    Suppose we define the ordered pair $(x, y)$ for any objects $x$ and $y$ by the formula $(x, y) \coloneqq \{\{x\}, \{x, y\}\}$.
    Show that such a definition indeed obeys the property (3.5), and also whenever $X$ and $Y$ are sets, the Cartesian product $X \times Y$ is also a set.
    Thus this definition can be validly used as a definition of an ordered pair.
    For an additional challenge, show that the alternate definition $(x, y) \coloneqq \{x, \{x, y\}\}$ also verifies (3.5) and is thus also an acceptable definition of ordered pair.
}
    Recall that (3.5) states two ordered pairs $(x, y)$ and $(x', y')$ are equal if and only if $x = x'$ and $y = y'$.
    Write $(x, y) = \{ \{ x \}, \{ x, y \} \}$, $(x', y') = \{ \{ x' \}, \{ x', y' \} \}$.

    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item Show that such a definition indeed obeys the property (3.5).

            The right-to-left implication is easy to prove:
            if $x = x'$ and $y = y'$, then $\{ \{ x \}, \{ x, y \} \} = \{ \{ x' \}, \{ x', y' \} \}$ by the substitution axiom (see Appendix A.7), thus $(x, y) = (x', y')$.
            Now consider the converse;
            we want to show if $\{ \{ x \}, \{ x, y \} \} = \{ \{ x' \}, \{ x', y' \}\}$, then $x = x'$ and $y = y'$.
            We will consider two cases.
            
            If $x = y$, then $\{ \{ x \}, \{ x, y \} \}$  simplifies to the singleton $\{\{ x \}\}$.
            Then, we have $\{ \{ x' \}, \{ x', y' \}\} = \{\{ x \}\}$.
            Therefore, $\{ \{ x' \}, \{ x', y' \}\}$ is also a singleton.
            A pair set is a singleton if and only if its elements are equal; therefore $\{ x' \} = \{ x', y' \}$, and thus $x' = y'$.
            We can also deduce $\{ x \} \in \{ \{ x' \}, \{ x', y' \}\}$, which leads to $x = x'$.
            By the hypothesis, we have $x = y$; thus we also have $y = y'$ because $x' = y'$, thus the property is true in the case that $x = y$.

            Consider instead the case where $x \ne y$.
            By the definition of equality of sets, we have $\{ x \} \in \{ \{ x' \}, \{ x', y' \}\}$, which implies $\{ x \} = \{ x' \}$ or $\{ x \} = \{ x', y' \}$.
            The latter is false as the left-hand side is a singleton while the right-hand side is a pair set.
            Thus, we have $\{ x \} = \{ x' \}$, or $x = x'$.
            Similarly, we have $\{ x, y \} \in \{ \{ x' \}, \{ x', y' \}\}$, which implies $\{ x, y \} = \{ x', y' \}$ as $\{ x, y \} \ne \{ x \}$.
            Therefore, $y \in \{ x', y'\}$, which leads to $y = x'$ or $y = y'$.
            Because $x = x'$ and $y \ne x$, we have $y \ne x'$, thus $y = y'$.
            We can now conclude that the property is true when $x \ne y$.
        
        \item Show that whenever $X$ and $Y$ are sets, the Cartesian product $X \times Y$ is also a set.

            Because $x \in X$ and $y \in Y$, $\{ x \}$ and $\{ x, y \}$ are subsets of $X \cup Y$.
            Therefore, both $\{ x \}$ and $\{ x, y \}$ are elements of $\mathcal{P}(X \cup Y)$.
            We can then construct the set $\{\{ x \}, \{ x, y \} \}$, which is a subset of $\mathcal{P}(X \cup Y)$, and thus an element of $\mathcal{P}(\mathcal{P}(X \cup Y))$.
            By the axiom of specification, we can construct a set $\{ S \in \mathcal{P}(\mathcal{P}(X \cup Y)) : S = \{\{x\}, \{x, y\}\} \text{ for some } x \in X, y \in Y \}$.
            Therefore, if an object $a$ is in this set, $a$ is an ordered set $\{\{x\}, \{x, y\}\}$ for some $x \in X$ and $y \in Y$.
            This is equivalent to the definition of the Cartesian product, $X \times Y$.

        \item Show that the alternate definition $(x, y) \coloneqq \{x, \{x, y\}\}$ also verifies (3.5).

            With the alternate definition, we want to show 
            \[
                 \{x, \{x, y\}\} = \{x', \{x', y'\}\} \iff x = x' \text{ and } y = y'.
            \]

            It is easy to prove the right-to-left implication:
            suppose $x = x'$ and $y = y'$; then $\{x, \{x, y\}\} = \{x', \{x', y'\}\}$ by the substitution axiom, thus $(x, y) = (x', y')$.
            
            Now consider the converse.
            Suppose $\{x, \{ x, y \}\} = \{x', \{x', y'\}\}$; we want to show $x = x'$ and $y = y'$.
            By the definition of equality of sets, we have $x \in \{ x', \{ x', y' \} \}$, thus $x = x'$ or $x = \{ x', y' \}$.
            Similarly, $x' \in \{ x, \{ x, y \} \}$, thus $x = x'$ or $x' = \{ x, y \}$.
            Suppose $x \ne x'$, thus $x = \{ x', y' \}$ and $x' = \{ x, y \}$.
            This means $x$ and $x'$ are members of each other: a direct contradiction to Exercise 3.2.2.
            Thus, we have $x = x'$.
            As $x = x'$ and $\{ x, y \} \ne x$ due to the axiom of regularity, we have $\{ x, y \} = \{ x', y' \}$.
            Assume $y = y'$.
            As $y \in \{ x', y' \}$ and $y' \in \{ x, y \}$, we have $y = x'$ and $y' = x$.
            Because $x = x'$, we have $y = y'$, a contradiction.
            Therefore, $y = y'$.
            Hence, the property (3.5) is valid with this definition of ordered sets.
    \end{enumerate}

    \begin{remark}
        My initial thought for (b) was to construct the set $X \times Y$ by the replacement axiom instead of looking for a master set.
        However, it is less connected with the given definitions of ordered sets.
        Nevertheless, I present the alternative solution:
    
        We first fix an arbitrary value of $y \in Y$, and let $x \in X$.
        Let $A$ be an arbitrary set.
        Because $y$ is fixed, there is exactly one $A$ such that $A = \{ x, \{ x, y \} \}$ by uniqueness of sets.
        Therefore, by the axiom of replacement, the set $\{A : A = \{x, \{x, y \}\} \text{ for some } x \in X \}$ exists.
        We simplify this set as $\{ \{ x, \{ x, y \} \} \text{ for some } x \in X \}$, and denote this set as $A_y$.
    
        Because for every element $y \in Y$ we have some set $A_y$, we can form the union set $\bigcup_{y \in Y} A_y = \bigcup \{A_y : y \in Y\} = \{ \{ x, \{ x, y \} \} \text{ for some } x \in X, y \in Y \}$.
    \end{remark}
\end{exer}


\begin{exer}{3.5.2}{
    Suppose we define an ordered $n$-tuple to be a surjective function $x : \{ i \in \mathbb{N} : 1 \le i \le n \} \to X$ whose range is some arbitrary set $X$ (so different ordered $n$-tuples are allowed to have different ranges);
    we then write $x_i$ for $x(i)$, and also write $x$ as $(x_i)_{1 \le i \le n}$.
    Using this definition, verify that we have $(x_i)_{1 \le i \le n} = (y_i)_{1 \le i \le n}$ if and only if $x_i = y_i$ for all $1 \le i \le n$.
    Also, show that if $(X_i)_{1 \le i \le n}$ is an ordered $n$-tuple of sets, then the Cartesian product, as defined in Definition 3.5.7, is indeed a set.
}
    For the first statement, we have 
    \begin{align*}
        (x_i)_{1 \le i \le n} = (y_i)_{1 \le i \le n}
        & \iff x(i) = y(i) \text{ for all } 1 \le i \le n \\
        & \iff x_i = y_i \text{ for all } 1 \le i \le n.
    \end{align*}
    Because ordered $n$-tuples are defined to be surjective, $x$ and $y$ have the same range.
    
    For the second statement, suppose $(X_i)_{1 \le i \le n}$ is an ordered $n$-tuple of sets.
    Let $I = \{ i \in \mathbb{N} : 1 \le i \le n \}$.
    From the union axiom (Axiom 3.11), the set $A = \bigcup \{X_i : 1 \le i \le n \}$ contains all elements from $X_1$ to $X_n$.
    From Exercise 3.4.7, the collection of all partial functions from $I$ to $A$ is a set.
    This set certainly contains all functions that have a domain of $I$ and range a subset of $A$.
    Using the axiom of specification, we can therefore produce the set $\{ x : I \to A : x(i) \in X_i \text{ for all } 1 \le i \le n \}$.
    This is equivalent to the set $\{ (x_i)_{1 \le i \le n}: x(i) \in X_i \text{ for all } 1 \le i \le n \}$, the definition as in Definition 3.5.7. \qed
\end{exer}


\begin{exer}{3.5.3}{
    Show that the definition of equality for ordered pair and ordered $n$-tuple obey the reflexivity, symmetry and transitivity axioms.
}
    We first show such properties for ordered pairs.
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item \textit{Reflexivity: $(x, y) = (x, y)$.}

            We have $x = x$ and $y = y$, which is true by reflexivity of objects.

        \item \textit{Symmetry: If $(x, y) = (x', y')$, then $(x', y') = (x, y)$.}

            We have $x = x'$ and $y = y'$ by the definition of equality of ordered tuples (Definition 3.5.1); thus $x' = x$, $y' = y$ by symmetry of objects.
            Therefore, applying Definition 3.5.1 again, we have $(x', y') = (x, y)$.

        \item \textit{Transitivity: If $(x, y) = (x', y')$ and $(x', y') = (x'', y'')$, then $(x, y) = (x'', y'')$.}

            The first equality gives $x = x'$, $y = y'$, while the second equality gives $x' = x''$ and $y' = y''$.
            By the transitivity of objects, we also have $x = x''$ and $y = y''$.
            Hence, using Definition 3.5.1, we have $(x, y) = (x'', y'')$.
    \end{enumerate}

    We will now show such properties for ordered $n$-tuples.
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item \textit{Reflexivity: $(x_i)_{1 \le i \le n} = (x_i)_{1 \le i \le n}$}
            
            By the definition of equality of ordered tuples (Definition 3.5.7), we have $x_i = x_i$ for all $1 \le i \le n$, which is true by reflexivity of objects.

        \item \textit{Symmetry: If $(x_i)_{1 \le i \le n} = (y_i)_{1 \le i \le n}$, then $(y_i)_{1 \le i \le n} = (x_i)_{1 \le i \le n}$.}

            For all $1 \le i \le n$, we have $x_i = y_i$ by definition of equality of ordered tuples (Definition 3.5.7).
            Thus, we have $y_i = x_i$ for all $1 \le i \le n$ due to the symmetry of objects.
            Applying Definition 3.5.7 again, we have $(y_i)_{1 \le i \le n} = (x_i)_{1 \le i \le n}$.

        \item \textit{Transitivity: If $(x_i)_{1 \le i \le n} = (y_i)_{1 \le i \le n}$ and $(y_i)_{1 \le i \le n} = (z_i)_{1 \le i \le n}$, then $(x_i)_{1 \le i \le n} = (z_i)_{1 \le i \le n}$.}

            According to the definition of equality of sets (Definition 3.5.7), we have $x_i = y_i$ and $y_i = z_i$ for all $1 \le i \le n$.
            Thus, by the transitivity of objects, we have $x_i = z_i$ for all $1 \le i \le n$.
            Using Definition 3.5.7 again, we obtain $(x_i)_{1 \le i \le n} = (z_i)_{1 \le i \le n}$. \qed
    \end{enumerate}    
\end{exer}


\begin{exer}{3.5.4}{
    Let $A$, $B$, $C$ be sets.
    Show that $A \times (B \times C) = (A \times B) \cup (A \times C)$,
    that $A \times (B \cap C) = (A \times B) \cap (A \times C)$,
    and that $A \times (B \setminus C) = (A \times B) \setminus (A \times C)$.
}
    First, we prove a simple lemma:
    \begin{lemma}
        An ordered set $(x, y)$ is an element of the Cartesian product $X \times Y$ if and only if $x \in X$ and $y \in Y$. In other words,
        \[
            (x, y) \in X \times Y \iff x \in X \text{ and } y \in Y.
        \]
    \end{lemma}
    \begin{prf}
        By definition, $X \times Y$ is the set of all ordered pairs $(x', y')$ for some $x' \in X$ and $y' \in Y$.
        Therefore, an ordered pair $(x, y)$ is an element of this set if and only if $x \in X$ and $y \in Y$.
    \end{prf}
    
    We want to show that if an ordered set $(x, y)$ is in either one of the set if and only if it is also in the other.
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item Show that $A \times (B \cup C) = (A \times B) \cup (A \times C)$.
            \begin{align*}
                (x, y) \in A \times (B \cup C)
                & \iff x \in A \text{ and } y \in B \cup C \\
                & \iff x \in A \text{ and } (y \in B \text{ or } y \in C) \\
                & \iff (x \in A \text{ and } y \in B) \text{ or } (x \in A \text{ and } y \in C) \\
                & \iff ((x, y) \in A \times B) \text{ or } ((x, y) \in A \times C) \\
                & \iff (x, y) \in (A \times B) \cup (A \times C).
            \end{align*}
        \item Show that $A \times (B \cap C) = (A \times B) \cap (A \times C)$.
            \begin{align*}
                (x, y) \in A \times (B \cap C)
                & \iff x \in A \text{ and } y \in B \cap C \\
                & \iff x \in A \text{ and } (y \in B \text{ and } y \in C) \\
                & \iff (x \in A \text{ and } y \in B) \text{ and } (x \in A \text{ and } y \in C) \\
                & \iff (x, y) \in A \times B \text{ and } (x, y) \in A \times C \\
                & \iff (x, y) \in (A \times B) \cap (A \times C)
            \end{align*}
        \item $A \times (B \setminus C) = (A \times B) \setminus (A \times C)$.
            \begin{align*}
                (x, y) \in A \times (B \setminus C)
                & \iff x \in A \text{ and } y \in B \setminus C \\
                & \iff x \in A \text{ and } (y \in B \text{ and } y \notin C) \\
                & \iff (x \in A \text{ and } y \in B) \text{ and } (x \in A \text{ and } y \notin C) \\
                & \iff (x, y) \in A \times B \text{ and } (x, y) \notin A \times C \\
                & \iff (x, y) \in (A \times B) \setminus (A \times C) \qed
            \end{align*}
    \end{enumerate}
\end{exer}



\begin{exer}{3.5.5}{
    Let $A$, $B$, $C$, $D$ be sets.
    Show that $(A \times B) \cap (C \times D) = (A \cap C) \times (B \cap D)$.
    Is it true that $(A \times B) \cup (C \times D) = (A \cup C) \times (B \cup D)$?
    Is it true that $(A \times B) \setminus (C \times D) = (A \setminus C) \times (B \setminus D)$?
}
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item Show that $(A \times B) \cap (C \times D) = (A \cap C) \times (B \cap D)$.
            \begin{align*}
                (x, y) \in (A \times B) \cap (C \times D)
                & \iff ((x, y) \in A \times B) \text{ and } ((x, y) \in C \times D) \\
                & \iff (x \in A \text{ and } y \in B) \text{ and } (x \in C \text{ and } y \in D) \\
                & \iff (x \in A \text{ and } x \in C) \text{ and } (y \in B \text{ and } y \in D) \\
                & \iff (x \in A \cap C) \text{ and } (y \in B \cap D) \\
                & \iff (x, y) \in (A \cap C) \times (B \cap D).
            \end{align*}
            
        \item Is it true that $(A \times B) \cup (C \times D) = (A \cup C) \times (B \cup D)$?
        
            This statement is false.
            Let $A = \{ 1 \}$, $B = \{ 2 \}$, $C = \{ 3 \}$, $D = \{ 4 \}$.
            Then, $(A \times B) \cup (C \times D) = \{(1, 2), (3, 4)\}$, but $(A \cup C) \times (B \cup D) = \{(1, 2), (1, 4), (3, 2), (3, 4)\}$.
            
        \item Is it true that $(A \times B) \setminus (C \times D) = (A \setminus C) \times (B \setminus D)$?
        
            This statement is also false.
            Let $A = \{ 1, 2 \}$, $B = \{ 3, 4 \}$, $C = \{ 2 \}$, $D = \{ 4 \}$.
            Then $(A \times B) \setminus (C \times D) = \{(1, 3), (1, 4), (2, 3)\}$, but $(A \setminus C) \times (B \setminus D) = \{(1, 3)\}$.
    \end{enumerate}
\end{exer}


\begin{exer}{3.5.6}{
    Let $A$, $B$, $C$, $D$ be non-empty sets.
    Show that $A \times B \subseteq C \times D$ if and only if $A \subseteq C$ and $B \subseteq C$,
    and that $A \times B = C \times D$ if and only if $A = C$ and $B = D$.
    What happens if the hypotheses that $A$, $B$, $C$, $D$ are all non-empty are removed?
}
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item Show that $A \times B \subseteq C \times D$ if and only if $A \subseteq C$ and $B \subseteq D$.
        
            Let $A$, $B$, $C$, $D$ be non-empty sets.
            We will first prove the left-to-right implication.
            Suppose $a \in A$, $b \in B$.
            Since $A \times B \subseteq C \times D$, we have $(a, b) \in C \times D$.
            By the lemma from the previous exercise, $a \in C$ and $b \in D$.
            Hence, $A \subseteq C$ and $B \subseteq D$.

            Now consider the converse.
            Suppose $A \subseteq C$ and $B \subseteq D$.
            Let $(a, b) \in A \times B$.
            Then, $a \in A$ and $b \in B$.
            Since $A \subseteq C$, $a \in C$.
            Since $B \subseteq D$, $b \in D$.
            Thus, we have $(a, b) \in C \times D$.
            This implies $A \times B \subseteq C \times D$.

        \item Show that $A \times B = C \times D$ if and only if $A = C$ and $B = D$.

            We will first prove the left-to-right implication.
            Suppose $A \times B = C \times D$; then we have $A \times B \subseteq C \times D$.
            From the first part of this exercise, this implies  $A \subseteq C$ and $B \subseteq D$.
            Similarly, $C \times D \subseteq A \times B$ implies $C \subseteq A$ and $D \subseteq B$.
            Hence, we have $A \times B = C \times D$ if $A = C$ and $B = D$.

            Now consider the converse.
            Suppose $A = C$ and $B = D$.
            Then, we have $A \subseteq C$ and $B \subseteq D$.
            From the first part of this exercise, this implies $A \times B \subseteq C \times D$.
            Similarly, $C \subseteq D$ and $B \subseteq A$ implies $C \times D \subseteq A \times B$.
            Hence, $A = C$ and $B = D$ if $A \times B = C \times D$.

        \item What happens if the hypotheses that $A$, $B$, $C$, $D$ are all non-empty are removed?

            The statements would then be false.
            Let $A = \{ 0 \}$, $B = C = D = \varnothing$.
            Then, $A \times B = C \times D = \varnothing$, but $A \not\subseteq C$. \qed
    \end{enumerate}
\end{exer}


\begin{exer}{3.5.7}{
    Let $X$, $Y$ be sets, and let $\pi_{X \times Y \to X} : X \times Y \to X$ and $\pi_{X \times Y \to Y} : X \times Y \to Y$ be the maps $\pi_{X \times Y \to X}(x, y) \coloneqq x$ and $\pi_{X \times Y \to Y} (x, y) \coloneqq y$;
    these maps are known as the \emph{co-ordinate functions} on $X \times Y$.
    Show that for any functions $f : Z \to X$ and $g : Z \to Y$ ,
    there exists a unique function $h : Z \to X \times Y$ such that $\pi_{X \times Y \to X} \circ h = f$ and $\pi_{X \times Y \to Y} \circ h = g$.
    This function h is known as the \emph{direct sum} of $f$ and $g$ and is denoted $h = f \oplus g$.
}
    We will first show the existence of the function $h$.
    Define $h$ as such: $h(z) = (f(z), g(z))$ for any $z \in Z$. Thus, the range of $h$ is $X \times Y$
    Thus, we have the following equalities:
    \[
        (\pi_{X \times Y \to X} \circ h)(z) = \pi_{X \times Y \to X}(h(x)) = \pi_{X \times Y \to X}((f(x), g(x)) = f(x)
    \]
    \[
        (\pi_{X \times Y \to Y} \circ h)(z) = \pi_{X \times Y \to Y}(h(x)) = \pi_{X \times Y \to Y}((f(x), g(x)) = g(x)
    \]
    Therefore, for this definition of $h$, the condition holds.

    We will now show the uniqueness of $h$.
    Suppose there is another function $h'$ such that $\pi_{X \times Y \to X} \circ h = f$ and $\pi_{X \times Y \to Y} \circ h = g$.
    For any $z \in Z$, $h'(z)$ must be an ordered set $(a, b)$.
    Then, we have 
    \[
        a = (\pi_{X \times Y \to X} \circ h')(z) = f(z)
        \qquad
        b = (\pi_{X \times Y \to Y} \circ h')(z) = g(z)
    \]
    Thus, $h'(z) = (f(z), g(z))$ for all $z \in Z$, which is exactly equal to our initial definition.
    Hence, the function $h$ is unique. \qed
\end{exer}


\begin{exer}{3.5.8}{
    Let $X_1, \ldots , X_n$ be sets.
    Show that the Cartesian product $\prod^n_{i=1}X_i$ is empty if and only if at least one of the $X_i$ is empty.
}
    We will first prove the left-to-right implication:
    If at least one of the $X_i$ is empty, then the Cartesian product $\prod^n_{i=1}X_i$ is empty.
    Suppose at least one of the $X_i$ is empty.
    Then, no objects are elements of that set.
    From the definition of Cartesian products, this implies there is no ordered set $(x_i)_{1 \le i \le n}$ such that $x_i \in X_i \text{ for all } 1 \le i \le n$.
    As such, the Cartesian product is empty.    
    
    Now consider the right-to-left implication:
    If the Cartesian product $\prod^n_{i=1}X_i$ is empty, then at least one of the $X_i$ is empty.
    If all $X_i$ are non-empty, then the Cartesian product will not be empty as per Lemma 3.5.12.
    Thus, if the Cartesian product is empty, at least one of the $X_i$ is empty. \qed
\end{exer}


\begin{exer}{3.5.9}{
    Suppose that $I$ and $J$ are two sets, and for all $\alpha \in I$ let $A_\alpha$ be a set, and for all $\beta \in J$ let $B_\beta$ be a set. Show that ($\bigcup_{\alpha \in I} A_\alpha) \cap (\bigcup_{\beta \in J} B_\beta) = \bigcup_{(\alpha, \beta) \in I \times J} (A_\alpha \cap B_\beta)$.
}
    We will first prove the left-to-right implication.
    Let $x \in (\bigcup_{\alpha \in I} A_\alpha) \cap (\bigcup_{\beta \in J} B_\beta)$.
    Then, $x \in \bigcup_{\alpha \in I} A_\alpha$ and $x \in \bigcup_{\beta \in J} B_\beta$.
    We have $x \in A_\alpha$ for some $\alpha \in I$ and $x \in B_\beta$ for some $\beta \in J$.
    Therefore, $x \in A_\alpha \cap B_\beta$ for some $\alpha \in I$ and $\beta \in J$.
    This is the same as saying $x \in A_\alpha \cap B_\beta$ for some $(\alpha, \beta) \in I \times J$.
    Hence, $x \in \bigcup_{(\alpha, \beta) \in I \times J} (A_\alpha \cap B_\beta)$.

    Now consider the converse.
    Let $x \in \bigcup_{(\alpha, \beta) \in I \times J} (A_\alpha \cap B_\beta)$.
    We have $x \in A_\alpha \cap B_\beta$ for some $(\alpha, \beta) \in I \times J$.
    Then, it follows that $x \in A_\alpha$ and $x \in \beta$ for some $\alpha \in I$ and $\beta \in J$.
    We can rewrite this as $x \in \alpha$ for some $\alpha in I$ and $x \in \beta$ for some $\beta \in J$.
    Thus, $x \in \bigcup_{\alpha \in I}$ and $x \in \bigcup_{\beta \in J}$, or $x \in (\bigcup_{\alpha \in I}) \cap (\bigcup_{\beta \in J})$.
\end{exer}


\begin{exer}{3.5.10}{
    If $f : X \to Y$ is a function, define the \emph{graph} of $f$ to be the subset of $X \times Y$ defined by $\{(x, f(x)) : x \in X\}$.
    Show that two functions $f : X \to Y$ , $\tilde{f} : X \to Y$ are equal if and only if they have the same graph.
    Conversely, if $G$ is any subset of $X \times Y$ with the property that for each $x \in X$, the set $\{y \in Y : (x, y) \in G \}$ has exactly one element (or in other words, $G$ obeys the \emph{vertical line test}), show that there is exactly one function $f : X \to Y$ whose graph is equal to $G$.
}
    We will first show the first part of the exercise.
    If $f = \tilde{f}$, then for any $x \in X$, $f(x) = \tilde{f}(x)$, and thus $(x, f(x)) = (x, \tilde{f}(x))$ for all $x$.
    Then the graphs of $f$ and $\tilde{f}$ are equal.
    Conversely, if the graphs of $f$ and $\tilde{f}$ are equal, then $(x, f(x)) = (x, \tilde{f}(x))$ for all $x \in X$.
    This implies $f(x) = \tilde{f}(x)$ for all $x$; it follows that $f = \tilde{f}$.

    Now consider the second part of the exercise.
    We will first show the existence of the function $f$.
    Denote the set $\{y \in Y : (x, y) \in G \}$ as $Y_x$.
    For each $x$, there is exactly one $y \in Y$ such that $(x, y) \in G$.
    Denote this $y$ as $y_x$.
    We can define $f : X \to Y$ such that $f(x) = y_x$.
    This is a valid function because for each $x \in X$, there is exactly one value of $y \in Y$.
    The graph of $f$ is $\{(x, f(x)) : x \in X \}$.
    If $(x, y) \in G$, then $y \in Y_x$. As $Y_x$ has exactly one element, it follows that $y = y_x = f(x)$.
    Thus, $(x, y) = (x, f(x))$, which is in the graph of $f$.
    Conversely, if $(x, y)$ is an element in the graph of $f$, then $y = f(x) = y_x$, and thus $y \in Y_x$, $(x, y) \in G$.
    Hence, the graph of $f$ is equal to $G$.

    We will now show the uniqueness of the function $f$.
    Suppose there is another function $h$ whose graph is equal to $G$.
    Then, the graph of $h$ is equal to the graph of $f$.
    From the first part of the exercise, this implies $h$ and $f$ are equal functions.
    Hence, the function $f$ is unique. \qed    
\end{exer}


\begin{exer}{3.5.11}{
    Show that Axiom 3.10 can in fact be deduced from Lemma 3.4.9 and the other axioms of set theory, and thus Lemma 3.4.9 can be used as an alternate formulation of the power set axiom.
}
    By the definition of Cartesian products (Definition 3.5.7), the Cartesian product $X \times Y$ is a set.
    From Lemma 3.4.9, the collection of all subsets of $X \times Y$ is also a set ($\mathcal{P}(X \times Y)$.
    Using the axiom of specification (Axiom 3.5), we can construct the following set:
    \[
        A = \{ G \in \mathcal{P}(X \times Y) : G \text{ obeys the vertical line test}\}.
    \]
    Note that $A \subseteq \mathcal{P}(X \times Y)$.
    According to Exercise 3.5.10, each element in $A$ is a graph.
    For each $G \in A$, there is exactly one $f$ whose graph is exactly equal to $G$.
    By the axiom of replacement (Axiom 3.6), we can now construct the following set:
    \[
        \{f : f \text{ has the graph } G \text{ for some } G \in \mathcal{P}(X \times Y) \}
    \]
    which is exactly equal to $Y^X$, the set of all functions with domain $X$ and range $Y$.
    This is exactly the definition of $Y^X$ in Axiom 3.10 (Power set axiom). \qed
\end{exer}

\begin{exer}{3.6.1}{
    Prove Proposition 3.6.4:
    Let $X$, $Y$, $Z$ be sets.
    Then $X$ has equal cardinality with $X$.
    If $X$ has equal cardinality with $Y$, then $Y$ has equal cardinality with $X$.
    If $X$ has equal cardinality with $Y$ and $Y$ has equal cardinality with $Z$, then $X$ has equal cardinality with $Z$.
}
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item \textit{$X$ has equal cardinality with $X$.}

            We can form the bijection $f : X \to X$ such that $f(x) = x$ for all $x \in X$.
            Thus, $X$ has equal cardinality with $X$.
            
        \item \textit{If $X$ has equal cardinality with $Y$, then $Y$ has equal cardinality with $X$.}
        
            If $X$ has equal cardinality with $Y$, there exists a bijection $f : X \to Y$.
            Then, we have the bijection $f^{-1} : Y \to X$, which implies $Y$ has equal cardinality with $X$.

        \item \textit{If $X$ has equal cardinality with $Y$ and $Y$ has equal cardinality with $Z$, then $X$ has equal cardinality with $Z$.}

            If $X$ has equal cardinality with $Y$, the bijection $f : X \to Y$ exists.
            Similarly, if $Y$ has equal cardinality with $Z$, the bijection $g : Y \to Z$ exists.
            Hence, we can construct the composition $g \circ f : X \to Z$, which is a bijection from $X$ to $Z$.
            Therefore, $X$ has equal cardinality with $Z$. \qed
    \end{enumerate}
\end{exer}


\begin{exer}{3.6.2}{
    Show that a set $X$ has cardinality $0$ if and only if $X$ is the empty set.
}
    First, consider the right-to-left implication.
    Suppose $X = \varnothing$; we want to show $\#(\varnothing) = 0$, or that the empty function $f : \varnothing \to \{i \in \mathbb{N} : 1 \le i \le 0 \}$ is a bijection.
    This is true because the range $\{i \in \mathbb{N} : 1 \le i \le 0 \}$ is an empty set, and from Exercise 3.3.3 the empty function is bijective when the range is also an empty set.

    We will now prove the left-to-right implication.
    Suppose a set $X$ has cardinality $0$.
    Then the function $f : \{i \in \mathbb{N} : 1 \le i \le 0 \} \to X$ is a bijection.
    Because the domain $\{i \in \mathbb{N} : 1 \le i \le 0 \}$ is an empty set, $f$ is an empty function.
    By Exercise 3.3.3, an empty function is bijective only when the range is an empty set.
    Hence, $X = \varnothing$. \qed    
\end{exer}


\begin{exer}{3.6.3}{
    Let $n$ be a natural number, and let $f : \{i \in \mathbb{N} : 1 \le i \le n \} \to \mathbb{N}$ be a function. Show that there exists a natural number $M$ such that $f(i) \le M$ for all $1 \le i \le n$.
}
    We apply induction on $n$.
    First, consider the base case $n = 1$ (the statement is vacuously true when $n = 0$).
    By choosing $M = \succr{f(1)}$, we see that indeed $f(i) \le M$ for all $i \in \{ 1 \}$.
    Thus, the statement is true for $n = 1$.

    Now, suppose inductively that the statement is true for some $n$; we want to show that it is still true for $\succr{n}$.
    We choose $M' =  M + f(\succr{n})$.
    By this definition, $M' \ge M$ because $f(\succr{n})$ is a natural number.
    This implies $M'$ is greater than all $f(i)$ for all $1 \le i \le n$ by the inductive hypothesis.
    Thus, $M' \ge f(i)$ for all $1 \le i \le n$ by transitivity.
    Also, $M' \ge f(\succr{n})$ because $M$ is a natural number.
    Therefore, $f(i) \le M'$ for all $1 \le i \le \succr{n}$ as desired.
    This closes the induction. \qed
\end{exer}


\begin{exer}{3.6.4}{
    Prove Proposition 3.6.14 (Cardinal arithmetic).
}
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item \textit{%
            Let $X$ be a finite set, and let $x$ be an object which is not an element of $X$.
            Then $X \cup \{ x \}$ is finite and $\#(X \cup \{ x \}) = \#(X) + 1$.
        }

            Because $X$ is finite, we have $\#(X) = n$, $n \in \mathbb{N}$.
            We can then form a bijection $f : \{ i \in \mathbb{N} : 1 \le i \le n \} \to X$.
            Define $g : \{ i \in \mathbb{N} : 1 \le i \le n + 1 \} \to (X \cup \{ x \})$ such that $g(m) = f(m)$ for $1 \le m \le n$, and $g(n + 1) = x$.
            As $x \notin X$, $g$ is a bijection.
            Thus, $\#(X \cup \{ x \}) = n + 1 = \#(X) + 1$.
            Because $X \cup \{ x \}$ has cardinality $n + 1$, it is also finite.
            
        \item \textit{%
            Let $X$ and $Y$ be finite sets.
            Then $X \cup Y$ is finite and $\#(X \cup Y ) \le \#(X) + \#(Y).$
            If in addition X and Y are disjoint (i.e., $X \cap Y = \varnothing$), then $\#(X \cup Y) = \#(X) + \#(Y)$.
        }

            Because $X$ and $Y$ are finite, we have $\#(X) = m$, $\#(Y) = n$, $m, n \in \mathbb{N}$.
            Then, we have the bijections $f : X \to \{ i \in \mathbb{N} : 1 \le i \le m \}$, $g : Y \to \{ i \in \mathbb{N} : 1 \le i \le n \}$.
            Define $h(p) = f(p)$ for $p \in X$, $h(p) = g(p) + m$ for $p \in Y, p \notin X$.
            Thus, $h$ maps bijectively from $X \cup Y$ to a subset of $\{ i \in \mathbb{N} : 1 \le i \le m + n \}$.
            Since every subset of $\{ i \in \mathbb{N} : 1 \le i \le m + n \}$ is finite, there exists $k \le m + n$ such that the subset has cardinality $k$.
            Hence, $\#(X \cup Y) = k$, and $\#(X \cup Y) \le \#(X) + \#(Y)$.
            
            In particular, if $X$ and $Y$ are disjoint, then $h$ maps bijectively from $X \cup Y$ to $\{ i \in \mathbb{N} : 1 \le i \le m +n \}$.
            Then, $\#(X \cup Y) = \#(X) + \#(Y)$ if $X$ and $Y$ are disjoint.

        \item \textit{%
            Let $X$ be a finite set, and let $Y$ be a subset of $X$.
            Then $Y$ is finite, and $\#(Y) \le \#(X)$.
            If in addition $Y \ne X$ (i.e., $Y$ is a proper subset of $X$), then we have $\#(Y) < \#(X)$.
        }

            Because $X$ is finite, we have $\#(X) = m$, $m \in \mathbb{N}$.
            We apply induction on $m$.
            For the base case, when $m = 0$, $X = \varnothing$.
            The only subset of $X$ is the empty set itself, which also has cardinality $0$.
            Thus, $Y$ is finite, and $\#(Y) \le \#(X)$.

            Now, assume inductively that $\#(Y) \le \#(X)$ for all $X$ with $m$ elements; we want to show this is also true for all $X'$ with $m + 1$ elements.
            Let $X' \coloneqq X \cup \{ x \}$ for some $x \notin X$, thus $\#(X') = \#(X) + 1$ as per results from (a).
            If $x \notin Y$, then $Y \subseteq X$, which has cardinality $n$, so the statement is true by the inductive hypothesis.
            If instead $x \in Y$, then $Y = (Y \setminus \{ x \}) \cup \{ x \}$.
            Because $Y \setminus \{ x \} \in X$, $Y \setminus \{ x \}$ is finite by the inductive hypothesis.
            Then, from results in (a), we have $\#(Y) = \#(Y \setminus \{ x \}) + 1$.
            Hence, $\#(Y) = \#(Y \setminus \{ x \}) + 1 \le \#(X) + 1 = \#(X')$.            

        \item \textit{%
            If $X$ is a finite set, and $f : X \to Y$ is a function, then $f(X)$ is a finite set with $\#(f(X)) \le \#(X)$.
            If in addition $f$ is one-to-one, then $\#(f(X)) = \#(X)$.
        }

            First suppose $f$ is one-to-one, then $f : X \to f(X)$ is a bijection; this implies $\#(f(X)) = \#(X)$.
            For the general case, we will induct on $m = \#(X)$.

            First, consider the base case $m = 0$.
            Then, $X$ has no elements, and therefore is empty.
            Thus, $f(X)$ is also empty, and thus has cardinality $0$ as per Exercise 3.6.2.
            Therefore, $\#(f(X)) \le \#(X)$.

            Now suppose $\#(f(X)) \le \#(X)$ for all $X$ with $m$ elements; we want to show that this is true for all $X'$ with $m + 1$ elements.
            Let $X' \coloneqq X \cup \{ x \}$ for some $x \notin X$, thus $\#(X') = \#(X) + 1$ as per results from (a).
            If $f(x) \in f(X)$, then $f(X) = f(X')$; we then have
            \[
                \#(f(X')) = \#(f(X)) \le \#(X) \le \#(X').
            \]
            If instead $f(x) \notin f(X)$, then $\#(f(X')) = \#(f(X) \cup \{f(x)\}) = \#(f(X)) + 1$.
            Because $\#(f(X)) \le \#(X)$,
            \[
                \#(f(X')) = \#(f(X)) + 1 \le \#(X) + 1 = \#(X').
            \]
            This closes the induction.
        
        \item \textit{%
            Let $X$ and $Y$ be finite sets.
            Then Cartesian product $X \times Y$ is finite and $\#(X \times Y) = \#(X) \times \#(Y)$.
        }
        
            Because $X$ and $Y$ are finite, we have $\#(X) = m$ and $\#(Y) = n$, $m, n \in \mathbb{N}$.
            We will induct on $n$, keeping $m$ fixed.
            
            Consider the base case $n = 0$, i.e. $Y$ is an empty set.
            Then $X \times Y$ is the empty set, and 
            \[
                \#(X \times Y) = 0 = 0m = mn
            \]
            as per Exercise 3.6.2.
            If $n = 1$, $Y = \{ y \}$ is a singleton set.
            We can form a bijection $f : X \to X \times Y$ by defining $f(x) = (x, y)$ for any $x \in X$.
            Therefore,
            \[
                \#(X \times Y) = \#(X) = m = 1m.
            \]
            Hence, the base case is proven.

            Now assume inductively that $\#(X \times Y) = \#(X) \times \#(Y)$ for all $Y$ with $n$ elements;
            We want to show that $\#(X \times Y) = \#(X) \times \#(Y)$ for all $Y$ with $n + 1$ elements.
            Let $y$ be an arbitrary element of $Y$, thus $Y \setminus \{ y \}$ has cardinality $n$ by Lemma 3.6.9.
            Note that $(Y \setminus \{ y \}) \cup \{ y \} = Y$, and $(Y \setminus \{ y \}) \cap \{ y \} = \varnothing$.
            By Exercise 3.5.4,
            \[
                X \times Y = (X \times (Y \setminus \{ y \})) \cup (X \times \{ y \}).
            \]
            As shown in the base case, $X \times \{ y \}$ has $m$ elements.
            Because $Y \setminus \{ y \}$ has $n$ elements, $X \times (Y \setminus \{ y \})$ has $mn$ elements by the inductive hypothesis.
            Using results from (b), we have
            \[
                \#(X \times Y) = \#(X \times (Y \setminus \{ y \})) + \#(X \times \{ y \}) = mn + m = m(n + 1)
            \]
            as desired. This closes the induction.
            
        \item \textit{%
            Let $X$ and $Y$ be finite sets.
            Then the set $Y^X$ (defined in Axiom 3.10) is finite and $\#(Y^X) = \#(Y)^{\#(X)}$.
        }
        
            Because $X$ and $Y$ are finite, we have $\#(X) = m$ and $\#(Y) = n$, $m, n \in \mathbb{N}$.
            We will induct on $m$, keeping $n$ fixed.

            If $m = 0$, then $X = \varnothing$ by Exercise 3.6.2.
            Then, there is only one function from $X$ to $Y$, namely the empty function.
            Thus, $Y^X$ has exactly $n^0 = 1$ element.
            Hence, the base case is true.

            Now suppose inductively $\#(Y^X) = n^m$ for all $X$ with $m$ elements;
            we want to show that $\#(Y^{X'}) = n^{m+1}$ for all $X'$ with $m+1$ elements.
            Let $X' \coloneqq X \cup \{ x \}$ for some $x \notin X$, thus $X = X' \setminus x$.
            Then, $X'$ has $m$ elements.
            From Exercise 3.5.10, each function $f \in Y^X$ has a graph, and this graph is unique to $f$.
            Therefore, using the axiom of replacement (Axiom 3.6), we will replace each $f \in Y^X$ with its graph, as defined in Exercise 3.5.10.
            Thus, we have the set
            \[
                S = \{G \in \mathcal{P}(X \times Y) : G \text{ is a graph for some } f \in Y^X \}.
            \]
            We can then construct the set of all graphs of functions from $X'$ to $Y$ by including the mapping from $x$ to some $y \in Y$:
            \[
                S' = \{G \cup \{(x, y)\} : G \in S, y \in Y \}.
            \]
            Thus, we have the mapping $h : (S \times Y) \to S'$ where $h(G, y) \coloneqq G \cup \{ (x, y) \}$.
            
            Suppose $h(G_1, y_1) = h(G_2, y_2)$; then $G_1 \cup \{ (x, y_1) \} = G_2 \cup \{ (x, y_2) \}$.
            Since $x \notin X$, the only ordered pair in both graphs with first element $x$ is the added element.
            Thus, by definition, $y_1 = y_2$.
            Removing that common pair yields $G_1 = G_2$ as well, hence $(G_1, y_1) = (G_2, y_2)$.
            This means $h$ is injective.
            Let $A \in S'$, then $A$ is a graph of some $a : X' \to Y$.
            Then, we have $B = A \setminus \{(x, a(x))\}$.
            Because $x \notin X'$, $B$ is a graph of some function $b : X' \to Y$.
            Thus, $B \in S$, and we have $h(B, a(x)) = B \cup \{ (x, a(x)) \} = A$.
            This means $h$ is surjective.
            As $h$ is both injective and surjective, $h$ is a bijection from $(S \times Y)$ to $S'$.
            
            We know $S$ has the same cardinality as $Y^X$, which has $n^m$ elements by the inductive hypothesis.
            Therefore, $S \times Y$ has $n^m \times n = n^{m + 1}$ elements as proven in (e), so $S^Y$ has cardinality $n^{m + 1}$ as desired.
            This closes the hypothesis. \qed
    \end{enumerate}
\end{exer}


\begin{exer}{3.6.5}{
    Let $A$ and $B$ be sets.
    Show that $A \times B$ and $B \times A$ have equal cardinality by constructing an explicit bijection between the two sets.
    Then use Proposition 3.6.14 to conclude an alternate proof of Lemma 2.3.2.
}
    Define $f : (A \times B) \to (B \times A)$ such that $f(a, b) = (b, a)$.
    To show $f$ is a bijection, we must demonstrate its injectivity and surjectivity.
    Suppose $f(a, b) = f(a', b')$; then $(b, a) = (b', a')$.
    It follows that $b = b', a = a'$, and thus $(a, b) = (a', b')$.
    Therefore, $f$ is injective.
    Also, for any $(b, a)$ such that $b \in B$ and $a \in A$, we can always find an ordered pair $(a, b) \in A \times B$ such that $f(a, b) = (b, a)$.
    This implies $f$ is also surjective.
    Because $f$ is both injective and surjective, we can conclude that $f$ is a bijection from $A \times B$ to $B \times A$; hence $A \times B$ and $B \times A$ have equal cardinality.

    Recall that Lemma 2.3.2 states that $n \times m = m \times n$ for any $m, n \in \mathbb{N}$.
    Let $M$ and $N$ be sets such that $\#(M) = m$, $\#(N) = n$.
    Then, $\#(M \times N) = m \times n$, $\#(N \times M) = n \times m$.
    However, $\#(N \times M) = \#(M \times N)$, so $m \times n = n \times m$ as required. \qed
\end{exer}


\begin{exer}{3.6.6}{
    Let $A$, $B$, $C$ be sets.
    Show that the sets $(A^B)^C$ and $A^{B \times C}$ have equal cardinality by constructing an explicit bijection between the two sets.
    Conclude that $(a^b)^c = a^{bc}$ for any natural numbers $a$, $b$, $c$.
    Use a similar argument to also conclude $a^b \times a^c = a^{b+c}$.
}
    The set $(A^B)^C$ is the set of all functions mapping from $C$ to $A^B$,
    and the set $A^{B \times C}$ is the set of all functions mapping from $B \times C$ to $A$.
    Then, for any $f \in (A^B)^C$, we have $f(\gamma) \in A^B$ for any $\gamma \in C$.
    Because $f(\gamma)$ is a function mapping from $B$ to $A$, we define $h : (A^B)^C \to A^{B \times C}$ such that $h(f)(\beta, \gamma) \coloneqq f(\gamma)(\beta)$ for any $\beta \in B, \gamma \in C$.

    Suppose $h(f) = h(\tilde{f})$ for some $\tilde{f} : C \to A^B$; then $h(f)(\beta, \gamma) = h(\tilde{f})(\beta, \gamma)$ for all $\beta \in B, \gamma \in C$.
    Thus, $f(\gamma)(\beta) = \tilde{f}(\gamma)(\beta)$, which leads to $f(\gamma) = \tilde{f}(\gamma)$, and thus $f = \tilde{f}$.
    This implies $h$ is injective.
    Suppose instead we have some $k \in A^{B \times C}$; thus $k : B \times C \to A$.
    We can find a $f \in (A^B)^C$ such that $f(\gamma)(\beta) = k(\beta, \gamma)$ for any $\beta \in B, \gamma \in C$.
    Indeed, for any fixed $\gamma \in C$, $\beta \mapsto k(\beta, \gamma)$ is a function mapping from $B$ to $A$.
    This implies $f(\gamma) \in A^B$, and therefore $f : C \to A^B$ is well-defined.
    For any $k$, we can find an $f$ such that $h(f) = k$; this means $h$ is surjective.
    Hence, $h$ is a bijection, and $\#((A^B)^C) = \#(A^{B \times C})$.

    Let $A, B, C$ be sets such that $\#(A) = a$, $\#(B) = b$, $\#(C) = c$.
    Then, by Proposition 3.6.14 (Cardinal arithmetic), $\#((A^B)^C) = (a^b)^c$, $ \#(A^{B \times C}) = a^{bc}$.
    However, $\#((A^B)^C) = \#(A^{B \times C})$, so $(a^b)^c = a^{bc}$ as required.

    Now consider the statement $a^b \times a^c = a^{b+c}$.
    For finite cardinalities $b$ and $c$, we can choose disjoint sets $B$ and $C$ such that $\#(B \cup C) = b + c$.
    Thus, to prove this statement, we want to show $\#(A^B \times A^C) = \#(A^{B \cup C})$ for some disjoint $B$ and $C$.
    We define $f : A^B \times A^C \to A^{B \cup C}$ such that $f(\beta, \gamma)(x) \coloneqq \beta(x)$ if $x \in B$ and $f(\beta, \gamma)(x) \coloneqq \gamma(x)$ if $x \in C$.

    Suppose $f(\beta, \gamma) = f(\tilde{\beta}, \tilde{\gamma})$ for some $\beta, \tilde{\beta} \in B$ and $\gamma, \tilde{\gamma} \in C$.
    Then, we have $f(\beta, \gamma)(x) = f(\tilde{\beta}, \tilde{\gamma})(x)$ for any $x \in B \cup C$.
    If $x \in B$, then the equation simplifies to $\beta(x) = \tilde{\beta}(x)$, which means $\beta = \tilde{\beta}$.
    Similarly, if $x \in C$, then $\gamma = \tilde{\gamma}$.
    Thus, $(\beta, \gamma) = (\tilde{\beta}, \tilde{\gamma})$, which means $f$ is injective.
    Now suppose instead we have $g \in A^{B \cup C}$; then we have a $\beta$ and $\gamma$ such that $g(x) = \beta(x)$ for $x \in B$ and $g(x) = \gamma(x)$ for $x \in C$.
    This means there is always a $(\beta, \gamma)$ such that $f(\beta, \gamma) = g$.
    This implies $f$ is surjective.    
    Then, the function $f$ is a bijection.
    Hence, we have $\#(A^B \times A^C) = \#(A^{B \cup C})$, and therefore $a^b \times a^c = a^{b+c}$ as required. \qed
\end{exer}


\begin{exer}{3.6.7}{
    Let $A$ and $B$ be sets.
    Let us say that $A$ has \emph{lesser or equal} cardinality to $B$ if there exists an injection $f : A \to B$ from $A$ to $B$.
    Show that if $A$ and $B$ are finite sets, then $A$ has lesser or equal cardinality to $B$ if and only if $\#(A) \le \#(B)$.
}
    Suppose $A$ and $B$ are finite sets.
    We will first prove the left-to-right implication.
    Suppose $A$ has lesser or equal cardinality to $B$; we want to show that $\#(A) \le \#(B)$.
    By definition of lesser or equal cardinality, there exists an injection $f : A \to B$.
    Thus, $f : A \to f(A)$ is a bijection, and since $f(A) \subseteq B$, $\#(A) = \#(f(A)) \le \#(B)$.

    Now consider the converse.
    Suppose $\#(A) \le \#(B)$; we want to show that $A$ has lesser or equal cardinality to $B$.
    We can find bijections $f : A \to \{ i \in \mathbb{N} : 1 \le i \le \#(A) \}$ and $g : B \to \{ i \in \mathbb{N} : 1 \le i \le \#(B) \}$.
    We define $h : \{ i \in \mathbb{N} : 1 \le i \le \#(A) \} \to \{ i \in \mathbb{N} : 1 \le i \le \#(B) \}$ such that $h(i) = i$.
    This is well-defined because every element of $\{ i \in \mathbb{N} : 1 \le i \le \#(A) \}$ is an element of $\{ i \in \mathbb{N} : 1 \le i \le \#(B) \}$.
    
    Then, the composition $k = g^{-1} \circ h \circ f$ maps $A$ to $B$.
    Suppose $k(x) = k(y)$ for some $x, y \in A$; then $g^{-1}(h(f(x))) = g^{-1}(h(f(y)))$.
    Applying $g$ on both sides produces $f(x) = f(y)$, and since $f$ is a bijection, we have $x = y$.
    Hence, $k$ is an injection from $A$ to $B$, or $A$ has lesser or equal cardinality to $B$.
\end{exer}


\begin{exer}{3.6.8}{
    Let $A$ and $B$ be sets such that there exists an injection $f : A \to B$ from $A$ to $B$ (i.e., $A$ has lesser or equal cardinality to $B$).
    Show that there then exists a surjection $g : B \to A$ from $B$ to $A$.
}
    If $A$ is empty, then the function $g : B \to A$ is automatically surjective.
    Now consider if $A$ is non-empty.
    Define $g : B \to A$ such that $g(x) \coloneqq f^{-1}(x)$ if $x \in f(A)$, and $g(x) \coloneqq a$ for some $a \in A$ if $x \in B \setminus f(A)$.
    To show $g$ is surjective, for any arbitrary $a \in A$, we have a $f(a) \in B$, where $g(f(a)) = f^{-1}(f(a)) = a$.
    Hence, there exists a surjection $g : B \to A$ from $B$ to $A$.
\end{exer}


\begin{exer}{3.6.9}{
    Let $A$ and $B$ be finite sets.
    Show that $A \cup B$ and $A \cap B$ are also finite sets, and that $\#(A) + \#(B) = \#(A \cup B) + \#(A \cap B)$.
}
    $A \cup B$ is finite by Proposition 3.6.14(b), while by Proposition 3.6.14(c), $A \cap B$ is finite as $A \cap B \subseteq A$.

    Notice that
    \[
        A = (A \cap B) \cup (A \setminus B) \quad \text{ and } \quad B = (A \cap B) \cup (B \setminus A).
    \]
    Also,
    \[
        (A \cap B) \cap (A \setminus B) = \varnothing \quad \text{ and } \quad (A \cap B) \cap (B \setminus A) = \varnothing.
    \]
    Thus, by Proposition 3.6.14(b), we have
    \[
        \#(A) = \#(A \cap B) + \#(A \setminus B) \quad \text{ and } \quad \#(B) = \#(A \cap B) + \#(B \setminus A).
    \]
    Adding the two equations together, we have
    \[
        \#(A) + \#(B) = \#(A \cap B) + \#(A \setminus B) + \#(A \cap B) + \#(B \setminus A).
    \]
    By Exercise 3.1.10, $(A \setminus B)$, $(A \cap B)$, and $(B \setminus A)$ are disjoint, and their union is $A \cup B$.
    Thus,
    \[
        \#(A \setminus B) + \#(A \cap B) + \#(B \setminus A) = \#(A \cup B).
    \]
    Substituting this into the previous equation, we have 
    \[
        \#(A) + \#(B) = \#(A \cup B) + \#(A \cap B)
    \]
    as required. \qed
    
\end{exer}


\begin{exer}{3.6.10}{
    Let $A_1, \ldots, A_n$ be finite sets such that $\#(\bigcup_{i \in \{1,\ldots,n\}} A_i) > n$.
    Show that there exists $i \in \{1, \ldots ,n \}$ such that $\#(A_i) \ge 2$.
    (This is known as the \emph{pigeonhole principle}.)
}
    Suppose, for the sake of contradiction, that $\#(A_i) < 2$ for all $i \in \{1, \ldots ,n \}$.
    Then, $\#(A_i) \le 1$ for all $i \in \{1, \ldots ,n \}$.
    By applying Proposition 3.6.14(b) $n$ times, we obtain 
    \[
        \# \left( \bigcup_{i \in \{1,\ldots,n\}} A_i \right) \le \sum^n_{i=1} \#(A_i) \le n,
    \]
    which is a contradiction.
    Thus, there exists $i \in \{1, \ldots ,n \}$ such that $\#(A_i) \ge 2$. \qed
\end{exer}

\begin{exer}{4.1.1}{
    Verify that the definition of equality on the integers is both reflexive and symmetric.
}
    Let $a \minus b$ and $c \minus d$ be arbitrary integers.
    
    To prove reflexivity, we want to show that $a \minus b = a \minus b$.
    By definition of integers, we then have $a + b = b + a$, which is true by commutativity of addition.
    Thus, this definition is reflexive.

    To prove symmetry, we want to show that if $a \minus b = c \minus d$, then $c \minus d = a \minus b$.
    Suppose $a \minus b = c \minus d$; then $a + d = c + b$.
    By symmetry of equality, we then have $c + b = a + d$, which implies $c \minus d = a \minus b$. \qed    
\end{exer}


\begin{exer}{4.1.2}{
    Show that the definition of negation on the integers is well-defined in the sense that if $(a \minus b) = (a' \minus b')$, then $-(a \minus b) = -(a' \minus b')$ (so equal integers have equal negations).
}
    Suppose $a \minus b = a' \minus b'$; then $a + b' = b + a'$.
    By definition of negation, $-(a \minus b) = b \minus a$ and $-(a' \minus b') = b' \minus a'$.
    To show $-(a \minus b) = -(a' \minus b')$, we will show $b \minus a = b' \minus a'$, or $b + a' = b' + a$.
    We have shown this is true by the hypothesis. \qed    
\end{exer}


\begin{exer}{4.1.3}{
    Show that $(-1) \times a = -a$ for every integer $a$.
}
    Suppose $a$ is an integer; then $a = m \minus n$ for natural numbers $m$ and $n$.
    By the definition of integer products and multiplication of natural numbers, we have
    \begin{align*}
        (-1) \times a
        &= (0 \minus 1)(m \minus n)
        = (0m + 1n) \minus (0n + 1m) \\
        &= n \minus m
        = -(m \minus n)
        = -a.
    \end{align*}
    Hence we have $(-1) \times a = -a$ as desired. \qed
\end{exer}


\begin{exer}{4.1.4}{
    Prove the remaining identities in Proposition 4.1.6 (Laws of algebra for integers).
}
    Let $x = a \minus b$, $y = c \minus d$, $z = e \minus f$.
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item $x + y = y + x$.
            \begin{align*}
                x + y
                &= (a \minus b) + (c \minus d)
                = (a + c) \minus (b + d) \\
                y + x
                &= (c \minus d) + (a \minus b)
                = (c + a) \minus (d + b)
                = (a + c) \minus (b + d)
            \end{align*}
            Thus $x + y = y + x$.
            
        \item $(x + y) + z = x + (y + z)$.
            \begin{align*}
                (x + y) + z
                &= ((a \minus b) + (c \minus d)) + (e \minus f) \\
                &= ((a + c) \minus (b + d)) + (e \minus f) \\
                &= (a + c + e) \minus (b + d + f) \\
                x + (y + z)
                &= (a \minus b) + ((c \minus d) + (e \minus f)) \\
                &= (a \minus b) + ((c + e) \minus (d + f)) \\
                &= (a + c + e) \minus (b + d + f)
            \end{align*}
            Therefore, $(x + y) + z = x + (y + z)$.
            
        \item $x + 0 = 0 + x = x$.
        
            We have previously shown commutativity of addition for integers, so $x + 0 = 0 + x$.
            Thus, we only need to show $x + 0 = x$:
            \[
                x + 0 = (a \minus b) + (0 \minus 0) = (a + 0) \minus (b + 0) = a \minus b = x.
            \]
            Hence, $x + 0 = 0 + x = x$.
            
        \item $x + (-x) = (-x) + x = 0$.
        
            We have previously shown commutativity of addition for integers, so $x + (-x) = (-x) + x$.
            Then, we only need to show $x + (-x) = 0$:
            \[
                x + (-x) = (a \minus b) + (b \minus a) = (a + b) \minus (b + a) = 0.
            \]
            It follows that $x + (-x) = (-x) + x = 0$.
            
        \item $xy = yx$.
            \begin{align*}
                xy &= (a \minus b) \times (c \minus d) = (ac + bd) \minus (ad + bc) \\
                yx &= (c \minus d) \times (a \minus b) = (ca + db) \minus (da + cb) = (ac + bd) \minus (ad + bc)
            \end{align*}
            Thus, $xy = yx$.

        \item $(xy)z = x(yz)$.

            The proof of this law is given by Tao.
            
        \item $x1 = 1x = x$.
        
            We have shown previously the commutativity of integer multiplication, so $x1 = 1x$.
            Thus, we only need to show $x1 = x$:
            \[
                x1 = (a \minus b)(1 \minus 0) = (a1 + b0) \minus (a0 + b1) = a \minus b = x.
            \]
            Therefore, $x1 = 1x = x$.
            
        \item $x(y + z) = xy + xz$.
            \begin{align*}
                x(y + z)
                &= (a \minus b)((c \minus d) + (e \minus f)) \\
                &= (a \minus b)((c + e) \minus (d + f)) \\
                &= (a(c + e) + b(d + f)) \minus (a(d + f) + b(c + e)) \\
                &= (ac + ae + bd + bf) \minus (ad + af + bc + be) \\
                xy + xz
                &= (a \minus b)(c \minus d) + (a \minus b)(e \minus f) \\
                &= ((ac + bd) \minus (ad + bc)) + ((ae + bf) \minus (af + be)) \\
                &= (ac + ae + bd + bf) \minus (ad + af + bc + be)
            \end{align*}
            This implies $x(y + z) = xy + xz$.
            
        \item $(y + z)x = yx + zx$.
            \begin{align*}
                (y + z)x
                &= x(y + z) \\
                yx + zx
                &= xy + xz
            \end{align*}
            By part (h), these are equivalent. We can conclude that $(y + z)x = yx + zx$. \qed
            
    \end{enumerate}
\end{exer}


\begin{exer}{4.1.5}{
    Prove Proposition 4.1.8 (Integers have no zero divisors):
    Let $a$ and $b$ be integers such that $ab = 0$.
    Then either $a = 0$ or $b = 0$ (or both).
}
    Let $m, p, q$ be positive natural numbers, and $b = p \minus q$.
    Suppose $ab = 0$ and $a \ne 0$; we will show this forces $b = 0$.
    By the trichotomy of integers (Lemma 4.1.5), we will split into three cases:
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item $a = 0$.
        
            This case contradicts our hypothesis.
            
        \item $a > 0$.

            Then, $a = m \minus 0$.
            \[
                ab = (m \minus 0)(p \minus q) = (mp + 0q)\minus (0p + mq) = mp \minus mq
            \]
            If $mp \minus mq = 0$, then $mp = mq$.
            Since $m$ is nonzero, $p = q$ by the cancellation law; thus $b = 0$.

        \item $a < 0$.

            Then $a = 0 \minus m$.
            \[
                ab = (0 \minus m)(p \minus q) = (0p + mq)\minus (mp + 0q) = mq \minus mp
            \]
            If $mq \minus mp = 0$, then $mq = mp$.
            Since $m$ is nonzero, $q = p$ by the cancellation law; thus $b = 0$.
    \end{enumerate}

    Hence if $ab = 0$ and $a \ne 0$, then $b = 0$. \qed
\end{exer}


\begin{exer}{4.1.6}{
    Prove Corollary 4.1.9 (Cancellation law for integers):
    If $a, b, c$ are integers such that $ac = bc$ and $c$ is non-zero, then $a = b$.
}
    We add $-bc$ to both sides of the equation to obtain
    \[
        ac + (-bc) = bc + (-bc).
    \]
    The right-hand side of the equation is equal to $0$.
    We use distributivity of integers to obtain the equation $(a - b)c = 0$.
    Because $c \ne 0$, $a - b = 0$ by Proposition 4.1.8.
    This implies $a = b$. \qed
\end{exer}


\begin{exer}{4.1.7}{
    Prove Lemma 4.1.11 (Properties of order).
}
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item $a > b$ if and only if $a - b$ is a positive natural number.

            Suppose $a > b$; then $a \ge b$ and $a \ne b$ by ordering of integers (Definition 4.1.10).
            This implies $a = b + n$ for some natural number $n$.
            By adding $-b$ to both sides (using the laws of algebra (Proposition 4.1.6)), it follows that $a - b = n$.
            Because $a \ne b$, $a - b \ne 0$, thus $n \ne 0$ by transitivity of equality.
            As $n$ is a natural number larger than $0$, $n$ is positive.
            Hence $a - b$ is a positive natural number.

            Now consider the converse.
            Denote $a - b = n$, and suppose $n$ is a positive natural number; thus $n > 0$.
            Then, $a = b + n$ by adding $b$ to both sides of the equation.
            As $n$ is an integer greater than $0$, we have $a > b$.
            
        \item (Addition preserves order) If $a > b$, then $a + c > b + c$.

            Suppose $a > b$; then $a = b + n$ for some positive natural number $n$.
            Adding $c$ to both sides of the equation, we have $a + c = b + c + n$.
            Using Definition 4.1.10 again, we have $a + c > b + c$.
        
        \item (Positive multiplication preserves order) If $a > b$ and $c$ is positive, then $ac > bc$.

            From (a), if $a > b$, then $a - b$ is positive.
            Because both $a - b$ and $c$ are positive, $(a - b)c$ is positive by Lemma 2.3.3.
            Using the laws of algebra and Exercise 4.1.3, we have
            \[
                (a - b)c = (a + (-b))c = ac + (-b)c = ac + (-1)bc = ac - bc.
            \]
            Then because $(a - b)c$ is positive, $ac - bc$ is also positive.
            Using (a) again gives $ac > bc$.
            
        \item (Negation reverses order) If $a > b$, then $-a < -b$.

            We will first introduce a short lemma: $-(-n) = n$.
            From the laws of algebra (Proposition 4.1.6), we have $n + (-n) = 0$.
            Adding the negation of $-n$ from both sides yield $n = 0-(-n) = -(-n)$.
            
            To show $-a < -b$, we need to show $(-b) - (-a) > 0$.
            The left-hand side simplifies to $(-b) - (-a) = -b + a = a - b$.
            However, from (a), if $a > b$, then $a - b$ is positive.
            
        \item (Order is transitive) If $a > b$ and $b > c$, then $a > c$.

            From (a), we have $a = b + m$, $b = c + n$ for some positive natural numbers $m, n$.
            Using the substitution axiom, it follows that $a = c + m + n$.
            Because $m + n$ is a positive natural number, $a > c$.
            
        \item (Order trichotomy) Exactly one of the statements $a > b$, $a < b$, or $a = b$ is true.

            Consider the integer $a - b$.
            By the trichotomy of integers (Lemma 4.1.5), exactly one of $a - b = 0$, $a - b > 0$, $a - b < 0$ is true.
            If $a - b = 0$, then $a = b$.
            If $a - b > 0$, then from (a) we know $a > b$.
            If $a - b < 0$, then $a - b = -n$ for some positive natural number $n$; it follows that $a + n = b$, and thus $a < b$.
            Hence, exactly one of $a > b$, $a < b$, or $a = b$ is true. \qed
    \end{enumerate}
\end{exer}


\begin{exer}{4.1.8}{
    Show that the principle of induction (Axiom 2.5) does not apply directly to the integers. 
    More precisely, give an example of a property $P(n)$ pertaining to an integer $n$ such that $P(0)$ is true, and that $P(n)$ implies $P(\succr{n})$ for all integers $n$, but that $P(n)$ is not true for all integers $n$.
    Thus induction is not as useful a tool for dealing with the integers as it is with the natural numbers.
}
    As a counterexample, let $P(x)$ be the property pertaining to an integer $x$ that $x \ge 0$.
    Then, $P(0)$ is true because $0 \ge 0$.
    Suppose inductively that $P(x)$ is true; then $\succr{x} \ge x \ge 0$, which means $P(\succr{x})$ is also true.
    However, $P(-1)$ is false because $-1 < 0$. \qed
\end{exer}


\subsection{The rationals}

\begin{exer}{4.2.1}{
    Show that the definition of equality for the rational numbers is reflexive, symmetric, and transitive.
}
    Let $a // b$, $c // d$, $e // f$ be rational numbers.
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item \textit{Reflexivity: $a // b = a // b$.}
            By definition, $a // b = a // b$ if $ab = ab$, which is obviously true.

        \item \textit{Symmetry: If $a // b = c // d$, then $c // d = a // b$.}
            Suppose $a // b = c // d$, then $ad = cb$.
            By commutativity of equality, we have $cb = ad$.
            From the definition of rational numbers, this implies $c // d  = a // b$.

        \item \textit{Transitivity: If $a // b = c // d$ and $c // d = e // f$, then $a // b = e // f$.}
            Suppose $a // b = c // d$ and $c // d = e // f$.
            Then $ad = cb$ and $cf = ed$.
            Multiplying the first equation by $f$ and the second equation by $b$, we have $adf = cbf$ and $cbf = edb$.
            Thus, we have $adf = edb$ by transitivity of equality.
            According to the cancellation law for integers (Corollary 4.1.9), since $d \ne 0$ by definition of rational numbers, we have $af = eb$.
            From the definition of rational numbers (Definition 4.2.1), this implies $a//b = e//f$. \qed
    \end{enumerate}
\end{exer}


\begin{exer}{4.2.2}{
    Prove the remaining components of Lemma 4.2.3:
    The sum, product, and negation operations on rational numbers are well-defined, in the sense that if one replaces $a // b$ with another rational number $a' // b'$ which is equal to $a // b$, then the output of the operations in Definition 4.2.2 remains unchanged, and similarly for $c // d$.
}
    Suppose $a // b = a' // b'$; then $ab' = ba'$.
    First consider the product operation.
    We want to show $(a // b) \times (c // d) = (a' // b') \times (c // d)$.
    By definition, the left-hand side is $(ac) // (bd)$ and the right-hand side is $(a'c) // (b'd)$.
    Thus, we want to show $(ac) // (bd) = (a'c) // (b'd)$, or $acb'd = a'cbd$.
    By the hypothesis, we have $ab' = a'b$, thus $ab'cd = a'cbd$ by multiplying $cd$ to the equation.
    A similar argument can be made for replacing $c // d$ and $c' // d'$.

    Now consider the negation operation.
    We want to show $-(a // b) = -(a' // b')$.
    By Definition 4.2.2, this means $(-a) // b = (-a') // b'$, and thus $-ab' = -a'b$.
    However, from multiplying $-1$ to both sides of the equation $ab' = ba'$ and the result of Exercise 4.1.3, we know $-ab' = -a'b$ is true. \qed
\end{exer}


\begin{exer}{4.2.3}{
     Prove the remaining components of Proposition 4.2.4.
}
    Let $x = a // b$, $y = c // d$, $z = e // f$.
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item $x + y = y + x$.
            \begin{align*}
                x + y
                &= (a // b) + (c // d)
                = (ad + bc) // (bd) \\
                y + x
                &= (c // d) + (a // b)
                = (cb + da) // (db)
                = (ad + bc) // (bd)
            \end{align*}
            Thus $x + y = y + x$.
            
        \item $(x + y) + z = x + (y + z)$.
            
            The proof of this law is given by Tao.
            
        \item $x + 0 = 0 + x = x$.
        
            We have previously shown commutativity of addition for rationals, so $x + 0 = 0 + x$.
            Thus, we only need to show $x + 0 = x$:
            \[
                x + 0 = (a // b) + (0 // 1) = (a1 + b0) // (b1) = a // b = x.
            \]
            Hence, $x + 0 = 0 + x = x$.
            
        \item $x + (-x) = (-x) + x = 0$.
        
            We have previously shown commutativity of addition for rationals, so $x + (-x) = (-x) + x$.
            Then, we only need to show $x + (-x) = 0$:
            \[
                x + (-x) = (a // b) + ((-a) // b) = (a + (-a)) // (bb) = 0 // (b^2).
            \]
            Note that $0 // (b^2) = 0 // 1 = 0$ because $0 \times 1 = 0 \times b^2$.
            Then, it follows that $x + (-x) = (-x) + x = 0$.
            
        \item $xy = yx$.
            \begin{align*}
                xy &= (a // b) \times (c // d) = (ac) // (bd) \\
                yx &= (c // d) \times (a // b) = (ca) // (db) = (ac) // (bd)
            \end{align*}
            Thus, $xy = yx$.

        \item $(xy)z = x(yz)$.
            \begin{align*}
                (xy)z &= ((a // b)(c // d))(e // f) = (ac // bd)(e // f) = (ace) // (bdf) \\
                x(yz) &= (a // b)((c // d)(e // f)) = (a // b)(ce // df) = (ace) // (bdf)
            \end{align*}
            Hence, $(xy)z = x(yz)$.
            
        \item $x1 = 1x = x$.
        
            We have shown previously the commutativity of rational multiplication, so $x1 = 1x$.
            Thus, we only need to show $x1 = x$:
            \[
                x1 = (a // b)(1 // 1) = (a1) // (b1) = a // b = x.
            \]
            Therefore, $x1 = 1x = x$.
            
        \item $x(y + z) = xy + xz$.
            \begin{align*}
                x(y + z)
                &= (a // b)((c // d) + (e // f)) \\
                &= (a // b)((cf + ed) // (df)) \\
                &= (acf + ade) // bdf \\
                xy + xz
                &= (a // b)(c // d) + (a // b)(e // f) \\
                &= (ac // bd) + (ae // bf) \\
                &= (acbf + bdae) // (bdbf)
            \end{align*}
            As $(acf + ade) b^2 df = (acbf + bdae)bdf$, this implies $x(y + z) = xy + xz$.
            
        \item $(y + z)x = yx + zx$.
            \begin{align*}
                (y + z)x
                &= x(y + z) \\
                yx + zx
                &= xy + xz
            \end{align*}
            By part (h), these are equivalent. We can conclude that $(y + z)x = yx + zx$. \qed
    \end{enumerate}
\end{exer}


\begin{exer}{4.2.4}{
    Prove Lemma 4.2.7:
    Let $x$ be a rational number.
    Then exactly one of the following three statements is true:
    (a) $x$ is equal to $0$.
    (b) $x$ is a positive rational number.
    (c) $x$ is a negative rational number.
}
    Let $x$ = $a / b$ for some integer $a$ and $b$ such that $b$ is positive.
    By the trichotomy of integers, exactly one of $a = 0$, $a > 0$, and $a < 0$ is true.
    If $a = 0$, then $x = 0$.
    By Definition 4.2.7, $x$ is positive if and only if $a > 0$.
    Similarly, $x$ is negative if and only if $a < 0$.
    Therefore, the three statements correspond to the three mutually exclusive cases of $a$.
    Hence, exactly one of the statements holds.
\end{exer}


\begin{exer}{4.2.5}{
    Prove Proposition 4.2.9.
}
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item (Order trichotomy) Exactly one of the three statements $x = y$, $x < y$, or $x > y$ is true.

            Consider the integer $x - y$.
            From Lemma 4.2.7, exactly one of $x - y$ being zero, positive, or negative is true.
            According to Definition 4.2.8, $x - y$ is positive if and only if $x > y$, and $x - y$ is negative if and only if $x < y$.
            Similarly, $x = y$ if and only if $x = y$.
            Thus, the three statements correspond to the three mutually exclusive cases of $x - y$.
            Hence, exactly one of the statements is true.
            
        \item (Order is anti-symmetric) One has $x < y$ if and only if $y > x$.

            Suppose $x < y$; then $x - y < 0$.
            Thus, $x - y = (-a) / b$ for some positive integers $a$ and $b$.
            We then have $y - x = -(x - y) = a / b$, a positive rational.
            Hence, $y > x$.

            Suppose instead $y > x$; then $y - x > 0$.
            Thus, $y - x = a / b$ for some positive integers $a$ and $b$.
            We then have $x - y = -(y - x) = (-a) / b$, a negative rational.
            Hence, $x < y$.
            
        \item (Order is transitive) If $x < y$ and $y < z$, then $x < z$.

            Suppose $x < y$ and $y < z$.
            Then, we have $x - y = (-a) / b$ and $y - z = (-c) / d$ for some positive integers $a, b, c, d$.
            It follows that
            \[
                x - z = (x - y) + (y - z) = (-a) / b + (-c) / d = (-ad -bc) / bd = -(ad + bc) / bd.
            \]
            Because $bd$ and $(ad + bc)$ are positive, $x - z$ is a negative rational.
            Hence, $x < z$.
            
        \item (Addition preserves order) If $x < y$, then $x + z < y + z$.

            Suppose $x < y$; then $x - y$ is negative.
            However, $(x + z) - (y + z) = x - y$; it follows that $(x + z) - (y + z)$ is negative, and thus $x + z < y + z$.
            
        \item (Positive multiplication preserves order) If $x < y$ and $z$ is positive, then $xz < yz$.

            Suppose $x < y$; then $x - y = (-a) / b$ for some positive integers $a$ and $b$.
            Suppose also $z$ is positive; write $z = c / d$ for positive integers $c$ and $d$.
            Then, $xz - yz = (x - y)z = ((-a) / b)(c / d) = (-ac) / bd$.
            Because $(a-c)$ is negative and $bd$ is positive, $xz - yz$ is negative.
            Hence, $xz < yz$. \qed
    \end{enumerate}
\end{exer}


\begin{exer}{4.2.6}{
    Show that if $x$, $y$, $z$ are rational numbers such that $x < y$ and $z$ is negative, then $xz > yz$.
}
    Suppose $x < y$; then $x - y = (-a) / b$ for some positive integers $a$ and $b$.
    Suppose also $z$ is negative; write $z = (-c) / d$. for positive integers $c$ and $d$.
    Then, $xz - yz = (x - y)z = ((-a) / b)((-c) / d) = ((-a)(-c)) / bd = ac / bd$.
    Because $ac$ and $bd$ are positive, $xz - yz$ is positive.
    Hence, $xz > yz$. \qed
\end{exer}


\subsection{Absolute value and exponentiation}

\begin{exer}{4.3.1}{
    Prove Proposition 4.3.3.
}
    Let $x$, $y$, $z$ be rational numbers.
    We will prove part (c) before part (b) to avoid circularity.
    (The order of proofs displayed here are adjusted accordingly.)
    
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item[(a)] (Non-degeneracy of absolute value) We have $\abs{x} \ge 0$. Also, $\abs{x} = 0$ if and only if $x$ is $0$.

            By the trichotomy of rationals (Lemma 4.2.7), exactly one of $x = 0$, $x > 0$, $x < 0$ is true.
            If $x = 0$, then $\abs{x} = 0 \ge 0$ by definition.
            If $x > 0$, then $\abs{x} = x$, which is positive.
            If $x < 0$, then $\abs{x} = -x > 0$.

        \item[(c)] We have the inequalities $-y \le x \le y$ if and only if $y \ge \abs{x}$. In particular, we have $-\abs{x} \le x \le \abs{x}$.

            First show the left-to-right implication.
            Suppose $-y \le x \le y$.
            We will split into cases.
            If $x \ge 0$, then $\abs{x} = x \le y$ by hypothesis.
            If $x < 0$, then $\abs{x} = -x$.
            If $-y \le x$, then $y \ge -x$ by multiplying both sides by $-1$ (Exercise 4.2.6).
            Because $\abs{x} = -x$, we have $y \ge \abs{x}$.

            Conversely, suppose $y \ge \abs{x}$.
            Note that from $(a)$, $\abs{x} \ge 0$.
            It follows that $y \ge 0$; and thus $-y \le 0$
            If $x = 0$, then $\abs{x} = 0$; hence $y \ge \abs{x}$.
            If $x > 0$, then $\abs{x} = x$; thus $y \ge x$ by hypothesis.
            Furthermore, since $x > 0 \ge -y$, we have $-y \le x \le y$.
            If $x < 0$, then $\abs{x} = -x$; thus $-y \le -x$ by hypothesis.
            Furthermore, since $x < 0 \le y$, we have $-y \le x \le y$ as desired.
            
            In particular, take $y = \abs{x}$; thus $\abs{x} \ge \abs{x}$.
            Then, we have $-\abs{x} \le x \le \abs{x}$ as desired.
            
        \item[(b)] (Triangle inequality for absolute value) We have $\abs{x+y} \le \abs{x} + \abs{y}$.

            We will split into multiple cases based on the trichotomy of rationals.
            If $x = 0$ or $y = 0$, then the inequality simplifies to $\abs{x} \le \abs{x}$ or $\abs{y} \ge \abs{y}$, which is always true.
            If $x, y \ge 0$, then $\abs{x + y} = x + y$ and $\abs{x} + \abs{y} = x + y$; both sides are equal.
            Similarly, if $x, y < 0$, then $\abs{x + y} = -x - y$ and $\abs{x} + \abs{y} = -x -y$; once again, both sides are equal.

            Now consider the case where $x$ and $y$ have opposite signs.
            Note that part (c) states $x \le \abs{x}$ for all rational $x$.
            Without loss of generality, let $x \ge 0$ and $y < 0$.
            The value $\abs{x + y}$ takes two cases.
            If $x + y \ge 0$, then $\abs{x + y} = x + y$.
            This is less than $\abs{x} + \abs{y}$ by part (c).
            If instead $x + y < 0$, then $\abs{x + y} = (-x) + (-y)$.
            By part (c), this is less than $\abs{-x} + \abs{-y} = \abs{x} + \abs{y}$.
            Hence, $\abs{x+y} \le \abs{x} + \abs{y}$.

        \item[(d)] (Multiplicativity of absolute value)
            We have $\abs{xy} = \abs{x}\abs{y}$.
            In particular, $\abs{-x} = \abs{x}$.

            If $x = 0$ or $y = 0$, both sides are equal to $0$.
            If $x, y > 0$, then $\abs{xy} = \abs{x}\abs{y} = xy$.
            If $x, y < 0$, then $xy$ is positive; thus $\abs{xy} = xy$.
            Also, $\abs{x}\abs{y} = (-x)(-y) = xy$.
            
            Finally, the case where $x$ and $y$ have opposite signs.
            Without loss of generality, take $x > 0$, $y < 0$.
            Then, $xy$ is negative, and $\abs{xy} = -xy$.
            Also, $\abs{x}\abs{y} = x(-y) = -xy$.

            In particular, take $y = -1$ to obtain $\abs{-x} = \abs{x}$.

        \item[(e)] (Non-degeneracy of distance)
            We have $d(x, y) \ge 0$.
            Also, $d(x, y) = 0$ if and only if $x = y$.

            By definition of distance, $d(x, y) = \abs{x - y}$.
            From (a), $\abs{x - y} \ge 0$; it follows that $d(x, y) \ge 0$.

            From (a), $\abs{x - y} = 0$ if and only if $x - y = 0$.
            In other words, we have $d(x, y) = 0$ if and only if $x = y$, as required.

        \item[(f)] (Symmetry of distance)
            $d(x, y) = d(y, x)$.

            Using the definition of distance and results from (d),
            \[
                d(x, y) = \abs{x - y} = \abs{-(x - y)} = \abs{y - x} = d(y, x).
            \]
            Hence, $d(x, y) = d(y, x)$ as desired.

        \item[(g)] (Triangle inequality for distance)
            $d(x, z) \le d(x, y) + d(y, z)$.

            Take $a = x - y$, $b = y - z$.
            \begin{align*}
                d(x, y) &= \abs{x - y} = \abs{a} \\
                d(y, z) &= \abs{y - z} = \abs{b} \\
                d(x, z) &= \abs{x - z} = \abs{(x - y) + (y - z)} = \abs{a + b}
            \end{align*}
            From (b), we have $\abs{a + b} \le \abs{a} + \abs{b}$; hence $d(x, z) \le d(x, y) + d(y, z)$. \qed
    \end{enumerate}
\end{exer}


\begin{exer}{4.3.2}{

\begin{exer}{4.3.3}{
    Prove Proposition 4.3.10.
}
    Let $x, y$ be rational numbers, and let $n, m$ be natural numbers.
    We prove part (b) last to avoid circularity.
    
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item[(a)] We have $x^n x^m = x^{n+m}$, $(x^n)^m = x^{nm}$, and $(xy)^n = x^n y^n$.
            \begin{enumerate}[topsep=0mm,label=(\roman*)]
                \item[(i)] $x^n x^m = x^{n+m}$.

                    We induct on $m$, fixing $n$.
                    First consider the base case $m = 0$.
                    Then,
                    \[
                        x^n x^0 = x^n (1) = x^n = x^{n + 0} = x^{n + m}.
                    \]
                    Now suppose inductively that $x^n x^m = x^{n+m}$ for some natural number $m$;
                    we want to show $x^n x^{m + 1} = x^{n+m+1}$
                    We have
                    \[
                        x^n x^{m + 1} = x^n x^m x = x^{n + m} x = x^{n + m + 1}.
                    \]
                    This closes the induction.
                
                \item[(ii)] $(x^n)^m = x^{nm}$.

                    We induct on $m$, fixing $n$.
                    When $m = 0$, we have $(x^m)^0 = 1$ and $x^{0m} = x^0 = 1$.
                    Now suppose inductively we have $(x^n)^m = x^{nm}$ for some natural number $m$; we want to show $(x^n)^{m+1} = x^{n(m + 1)}$.
                    We have
                    \[
                        (x^n)^{m+1} = (x^n)^m \times x^n = x^{nm} \times x^n = x^{nm + n} = x^{n(m+1)}.
                    \]
                    This closes the induction.
                
                \item[(iii)] $(xy)^n = x^n y^n$.

                    We induct on $n$.
                    If $n = 0$, then $(xy)^0 = 1$, and $x^0 y^0 = 1 \times 1 = 1$.
                    Now suppose inductively we have $(xy)^n = x^n y^n$ for some natural number $n$;
                    we want to show $(xy)^{n+1} = x^{n+1} y^{n+1}$.
                    We have
                    \[
                        (xy)^{n+1} = (xy)^n (xy) = (x^n x)(y^n y) = x^{n+1} y^{n+1}.
                    \]
                    This closes the induction.
            \end{enumerate}

        \item[(c)] If $x \ge y \ge 0$, then $x^n \ge y^n \ge 0$. If $x > y \ge 0$ and $n > 0$, then $x^n > y^n \ge 0$.

            We prove the latter statement first.
            Suppose $x > y \ge 0$.
            We induct on $n$.
            For the base case $n = 1$, $x^1 = x$, $y^1 = y$; hence, $x^1 > y^1 \ge 0$ is equivalent to $x > y \ge 0$, which is exactly the hypothesis.
            
            Now suppose inductively that $x^n \ge y^n \ge 0$ for some $n$;
            we want to show $x^{n+1} > y^{n+1} \ge 0$.
            Multiplying $x^n > y^n \ge 0$ by $y$, we have $x^n y > y^{n+1} \ge 0$.
            Multiplying $x > y \ge 0$ by $x^n$, we have $x^{n+1} > x^n y \ge 0$.
            Combining these inequalities, we have $x^{n+1} > x^n y > y^{n+1} \ge 0$.
            This closes the induction.

            We now show the former statement.
            Suppose $x \ge y \ge 0$; then $x = y$ or $x > y$.
            For $n = 0$, we have $x^0 = y^0 = 1 > 0$.
            If $x = y$, then $x^n = y^n > 0$.
            The case $x > y$ is exactly as proven above.
            Hence, if $x \ge y \ge 0$, then $x^n \ge y^n \ge 0$.

        \item[(d)] We have $\abs{x^n} = \abs{x}^n$.

            We induct on $n$.
            When $n = 0$, $\abs{x^0} = \abs{1} = 1$, and $\abs{x}^0 = 1$.
            Now suppose inductively that $\abs{x^n} = \abs{x}^n$; we need to show $\abs{x^{n+1}} = \abs{x}^{n+1}$.
            By Proposition 4.3.3(d) and the hypothesis, we have
            \[
                \abs{x^{n+1}} = \abs{x^n x} = \abs{x^n}\abs{x} = \abs{x}^n \abs{x} = \abs{x}^{n+1}.
            \]
            This closes the induction. \qed
            
        \item[(b)] Suppose $n > 0$. Then we have $x^n = 0$ if and only if $x = 0$.

            For the left-to-right implication, suppose for the sake of contradiction that there is a $x \ne 0$ such that $x^n = 0$.
            Then, we obtain $\abs{x} > 0$ as per Proposition 4.3.3(a).
            Applying Proposition 4.3.3(d) $n$ times, we have $\abs{x}^n = \abs{x^n}$.
            Since $\abs{x} > 0$, $\abs{x}^n > 0$ (by part (c)), hence $\abs{x^n} > 0$ (by part (d)), which contradicts the hypothesis.
            Hence, if $x^n = 0$, then $x = 0$.

            We now prove the right-to-left implication ($x^n = 0$ if $x = 0$).
            Suppose $x = 0$; we then have
            \[
                0^n = 0 \times 0 \times \ldots \times 0 = 0.
            \]
            This closes the induction.
            Hence, $x^n = 0$ if and only if $x = 0$.
        
    \end{enumerate}
\end{exer}


\begin{exer}{4.3.4}{
    Prove Proposition 4.3.12.
}
    For (a), (c), and (d), the statements have been proven for positive $m, n$.
    Let $x, y$ be non-zero rational numbers, and let $m, n$ be integers.
    Write $p = -m$, $q = -n$, so $p > 0$ if $m < 0$ and $q > 0$ if $n < 0$.
    
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item We have $x^n x^m = x^{n+m}$, $(x^n)^m = x^{nm}$, and $(xy)^n = x^n y^n$.

        \begin{enumerate}[topsep=0mm,label=(\roman*)]
            \item $x^n x^m = x^{n+m}$.

                We will split into cases.
                If $m \ge 0$, $n \ge 0$, then this is the case in Exercise 4.3.3.
                If $m, n < 0$, then
                \[
                    x^n x^m = x^{-q} x^{-p} = (1/x^q)(1/x^p) = 1/(x^q x^p) = 1/(x^{q+p}) = x^{-q-p} = x^{n+m}.
                \]
                Finally, the case where $m$ and $n$ have opposite signs.
                Without loss of generality, let $m \ge 0$, $n < 0$.
                Then,
                \[
                    x^n x^m = x^{-q} x^m = (1 / x^q)(x^m / 1) = x^m / x^q.
                \]
                If $m - q > 0$, then $x^{m-q} x^q = x^m$ by Proposition 4.3.10(a).
                Multiply $x^{-q}$ to the equation to get $x^{m-q} = x^m x^{-q}$, which we rewrite as $x^{m+n} = x^m x^n$ as required.
                Similarly, if $q - m > 0$, then $x^{q-m} x^m = x^q$.
                Multiply $1/(x^q x^{q-m})$ to the equation to get
                \begin{align*}
                    x^{q - m} x^m / x^q x^{q - m} &= x^q / x^q x^{q - m} & (\text{Multiply by } 1/(x^q x^{q-m})) \\
                    x^m / x^q &= 1 / x^{q - m} & (\text{Simplify})\\
                    x^m x^{-q} &= x^{m - q} & (\text{Definition 4.3.11})\\
                    x^m x^n &= x^{m + n}. & (\text{Substituting } n = -q)
                \end{align*}

                Hence, we have $x^n x^m$ for all integers $m, n$.
            
            \item $(x^n)^m = x^{nm}$.

                We first introduce a lemma: $(1/x^a)^b = x^{-ab}$ for positive $a, b$. We have
                \[
                    (1/x^a)^b (x^a)^b = (x^a / x^a)^b = 1^a = 1,
                \]
                then
                \[
                    (1/x^a)^b = 1/(x^a)^b = 1/x^{ab} = x^{-ab},
                \]
                hence proven.
                
                We will split into cases.
                If $m \ge 0$, $n \ge 0$, then this is the case in Exercise 4.3.3.
                If $m < 0$, $n \ge 0$, then
                \[
                    (x^n)^m = (x^n)^{-p} = (1/x^n)^p = x^{-np} = x^{n(-p)} = x^{mn}.
                \]
                For the case $m \ge 0$, $n < 0$, we have
                \[
                    (x^n)^m = (x^{-q})^m = (1/x^q)^m = x^{-qm} = x^{nm}.
                \]
                If $m, n < 0$, then
                \[
                    (x^n)^m = (x^{-q})^{-p} = 1 / (x^{-q})^p = 1 / x^{-pq} = x^pq = x^{(-m)(-n)} = x^{mn}.
                \]
            
            \item $(xy)^n = x^n y^n$.

                We will only prove the case of $n < 0$ as we have proven this statement for $n \ge 0$ in Exercise 4.3.3.
                \[
                    (xy)^n = 1/(xy)^q = 1/(x^q y^q) = 1/x^q \times 1/y^q = x^{-q} y^{-q} = x^n y^n.
                \]
            \end{enumerate}
        
        \item If $x \ge y > 0$, then $x^n \ge y^n > 0$ if $n$ is positive, and $0 < x^n \le y^n$ if $n$ is negative.

            From Proposition 4.3.10(c), we have $x^n \ge y^n \ge 0$ for positive $n$, so we only need to show $y^n \ne 0$ for non-zero $y$.
            This is true because the contrapositive of Proposition 4.3.10(b) states if $y \ne 0$, then $y^n \ne 0$.

            We now prove $0 < x^n \le y^n$ if $n$ is negative.
            Suppose $x \ge y > 0$; then $1/xy > 0$.
            We multiply $1/xy$ to both sides of $x \ge y$ to obtain
            \begin{align*}
                x &\ge y \\
                x/xy &\ge y/xy \\
                1/y &\ge 1/x
            \end{align*}

            Since $1/y \ge 1/x > 0$, we can apply previous results to show
            \begin{align*}
                (1/y)^{-n} \ge (1/x)^{-n} &> 0 \\
                y^{-(-n)} \ge x^{-(-n)} &> 0 \\
                y^n \ge x^n &> 0
            \end{align*}
            as required.

        \item If $x, y > 0$, $n \ne 0$, and $x^n = y^n$, then $x = y$.

            Suppose for the sake of contradiction that there exist $x, y> 0$ such that $x \ne y$ and $x^n = y^n$.
            Then, either $x > y$ or $x < y$.
            First consider $n > 0$.
            From part (b), if $x > y \ge 0$, then $x^n > y^n > 0$, which contradicts the hypothesis $x^n = y^n$.
            Similarly, if $y > x \ge 0$, then $y^n > x^n > 0$, which is also contradictory.
            Hence, we have shown $x = y$ for the case $n > 0$.
            
            Now consider $n < 0$.
            Then, because the reciprocal is well-defined, we have $1/x^n = 1/y^n$, which implies $x^{-n} = y^{-n}$.
            Because $n < 0$, $-n > 0$, which falls into the previous case.
            Hence, if $x, y > 0$, $n \ne 0$, and $x^n = y^n$, then $x = y$.

        \item We have $\abs{x^n} = \abs{x}^n$.

            We first introduce a small lemma: $\abs{1/a} = 1/\abs{a}$ for some rational $a$.
            By definition of rationals, $a \ne 0$.
            If $a > 0$, then $1/a > 0$; therefore $\abs{1/a} = 1/a = 1/\abs{a}$.
            If instead $a < 0$, $1/a < 0$, we then have
            \begin{align*}
                \abs{1/a} &= -1/a \\
                1/\abs{a} &= 1/-a = -1/a.
            \end{align*}
            We can now prove the statement.
            The case where $n \ge 0$ is proven in Proposition 4.3.10(d); we will only consider the case $n < 0$.
            We have
            \[
                \abs{x^n} = \abs{(1/x)^{-n}} = \abs{(1/x)^p} = \abs{1/x}^p = 1/\abs{x}^p = \abs{x}^{-p}
            \]
            as required. \qed
    \end{enumerate}
\end{exer}


\begin{exer}{4.3.5}{
    Prove that $2^N \ge N$ for all positive integers $N$.
}
    First consider the base case, $N = 1$.
    Then, $2^1 = 2 \ge 1$.
    Therefore, the base case is true.

    Now suppose inductively that $2^N \ge N$ for some positive integer $N$; we want to show that $2^{N+1} \ge N + 1$.
    From $N \ge 1$, we obtain $2N \ge N + 1$.
    Thus, we have 
    \[
        2^{N + 1} = 2 \times 2^N \ge 2 \times N \ge N + 1.
    \]
    as desired. \qed
\end{exer}


\subsection{Gaps in the rational numbers}

\begin{exer}{4.4.1}{
    Prove Proposition 4.4.1:
    Let $x$ be a rational number.
    Then there exists an integer $n$ such that $n \le x < n+1$.
    In fact, this integer is unique (i.e., for each $x$ there is only one $n$ for which $n \le x \le n + 1$).
    In particular, there exists a natural number $N$ such that $N > x$ (i.e., there is no such thing as a rational number which is larger than all the natural numbers).
}
    Let $x$ be a rational number, denote $x$ as $a / b$, such that $a, b$ are integers and $b > 0$.
    We first prove existence.
    First suppose $x \ge 0$.
    Then, $a \ge 0$; it follows that $a$ is a natural number.
    By Proposition 2.3.9 (Euclidean algorithm), there exist natural numbers $m, r$ such that $0 \le r < b$ and $a = mb + r$.
    Because $0 \le r < b$, $mb \le mb + r < mb + b$; this implies $mb \le a < (m+1)b$.
    Dividing the inequality by $b$ preserves order (because $b > 0$), thus $m \le a / b = x < m+1$.
    As $m$ is an integer, we have found an integer $m$ such that $m \le x < m+1$.

    Now suppose $x < 0$; then $-x$ is a positive rational.
    We have previously proven there exists an integer $m$ such that $m \le -x < m+1$.
    Multiplying the inequality by $-1$ reverses order; hence $-(m + 1) < x \le -m$.
    If $x \ne -m$, then $-(m + 1) \le x < -m$; we will therefore take $n \coloneqq -(m + 1)$.
    If instead $x = -m$, then $-m \le x < -m + 1$; hence we take $n \coloneqq -m$.

    We will now prove uniqueness.
    For the sake of contradiction, suppose there exist $m, m'$ such that $m \ne m'$, $m \le x < m + 1$ and $m' \le x < m' + 1$.
    Since $m \ne m'$, we have $m > m'$ or $m < m'$; we assume $m > m'$ without loss of generality.
    By Proposition 2.2.12(e), $m > m'$ if and only if $m \ge m' + 1$.
    Because $x < m' + 1$ by hypothesis, $x < m$, which contradicts the hypothesis.
    Hence, the integer is unique.

    To prove there exists a natural number $N$ such that $N > x$, we will split into cases.
    If $x < 0$, then take $N \coloneqq 0$.
    If $x > 0$, then there exists a natural number $n$ such that $n \le x < n + 1$ (because $n$ is a positive integer); hence, we can take $N \coloneqq n + 1$. \qed
\end{exer}


\begin{exer}{4.4.2}{
    A definition: a sequence $a_0, a_1, a_2, \ldots$ of numbers (natural numbers, integers, rationals, or reals) is said to be in infinite descent if we have $a_n > a_{n+1}$ for all natural numbers $n$ (i.e., $a_0 > a_1 > a_2 > \ldots$).
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item Prove the \emph{principle of infinite descent}:
            that it is not possible to have a sequence of natural numbers which is in infinite descent.
        \item Does the principle of infinite descent work if the sequence $a_1, a_2, a_3, \ldots$ is allowed to take integer values instead of natural number values?
            What about if it is allowed to take positive rational values instead of natural numbers? Explain.
    \end{enumerate}
}
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item For the sake of contradiction, assume there exists a sequence of natural numbers $a_0, a_1, a_2, \ldots$ such that $a_n > a_{n + 1}$ for all natural numbers $n$ (i.e., $a_0, a_1, a_2, \ldots$ is in infinite descent).
        Let $P(k)$ be a property pertaining to a natural number $k$ that $a_n \ge k$ for all $n \in \mathbb{N}$.
        We will show inductively that $P(k)$ is true for all $k \in \mathbb{N}$, and show that this is actually contradictory.
        
        The base case $k = 0$ is true because all $a_n$ are natural numbers, and thus $a_n \ge 0$ for all $n$.
        Now suppose inductively that we have all $a_n \ge k$ for some natural number $k$, 
        we want to show that $a_n \ge k + 1$.
        If $k = a_m$ for some $m \in \mathbb{N}$, then $k = a_m > a_{m + 1}$.
        However, since $m + 1$ is also a natural number, we have $a_{m + 1} > k$ by the hypothesis.
        This is clearly preposterous; thus we have $a_n > k$ for all $n \in \mathbb{N}$.
        By Proposition 2.2.12(e), we then have $a_n \ge k + 1$ for all $n \in \mathbb{N}$.
        This closes the induction.

        However, by taking $k = 0$ and $n = 1$, we have shown $a_1 \ge a_0$, which contradicts the hypothesis $a_0 > a_1 > \ldots$.
        Hence, the hypothesis is false, and no sequence of natural numbers in infinite descent exists.

        \item The principle of infinite descent does not work for integer values.
        As a counterexample, we define $a_n \coloneqq -n$.
        Because $-n > -(n + 1)$, we have $a_n > a_{n + 1}$ for all $n \in \mathbb{N}$.

        Similarly, for positive rationals, the principle of infinite descent does not work.
        As a counterexample, we define $a_n \coloneqq 1/n$.
        This is because $1/n > 1/(n+1)$ for all $n \in \mathbb{N}$ as shown by
        \[
            1/n - 1/(n+1) = 1/n + (-1/(n+1)) = (n+1 - n)/(n(n+1)) = 1/(n(n+1))> 0,
        \]
        hence $a_n > a_{n+1}$ for all $n \in \mathbb{N}$. \qed        
    \end{enumerate}
\end{exer}


\begin{exer}{4.4.3}{
    Fill in the gaps marked (why?) in the proof of Proposition 4.4.4.
}
    \begin{enumerate}[topsep=0mm,label=(\alph*)]
        \item \textit{Every natural number is either even or odd, but not both.}

            We first prove that every natural number is either even or odd.
            Assume $n$ is a natural number.
            By taking $q = 2$ in Proposition 2.3.9, we have $n = 2m + r$ and $0 \le r < 2$.
            In other words, $r = 0$ or $r = 1$.
            If $r = 0$, then $n$ is even; if $r = 1$, then $n$ is odd.

            We now prove that natural numbers cannot be both even and odd.
            Suppose for the sake of contradiction that there exists an $n \in \mathbb{N}$ that is both even and odd.
            Then $n = 2r + 1 = 2s$ for some natural numbers $r, s$, or $2(r - s) + 1 = 0$.
            If $r - s \ge 0$, then $2(r - s) + 1 \ge 2(0) + 1 = 1 > 0$.
            If $r - s < 0$, then $s - r > 0$.
            Because $s - r$ is a natural number, $s - r \ge 1$ by Proposition 2.2.12(e); hence $r - s \le -1$ by multiplying $-1$ to the inequality.
            Then, $2(r - s) + 1 \le 2(-1) + 1 = -1 < 0$.
            Hence, no natural number is both even and odd.            

        \item \textit{If $p$ is odd, then $p^2$ is also odd.}

            Suppose $p$ is odd; thus $p = 2k + 1$ for some $k \in \mathbb{N}$.
            We then have
            \[
                p^2 = (2k + 1)^2 = 4k^2 + 4k + 1 = 2(2k^2 + 2k) + 1.
            \]
            Since $2k^2 + 2k$ is a natural number, $p^2$ is odd.

        \item \textit{Since $p^2 = 2q^2$, we have $q < p$.}

            Suppose for the sake of contradiction that we have a $q \ge p$ such that $p^2 = 2q^2$.
            By Proposition 4.3.10(c), if $q \ge p > 0$, then $q^2 \ge p^2 > 0$.
            Since $2q^2 > q^2$, we have $2q^2 > p^2$, which contradicts the hypothesis.
            Hence, if $p^2 = 2q^2$, then $q < p$. \qed
            
    \end{enumerate}
\end{exer}

\subsection{Cauchy sequences}

\begin{exer}{5.1.1}{
    Prove Lemma 5.1.15:
    Every Cauchy sequence $(a_n)^\infty_{n=1}$ is bounded.
}
    Suppose $(a_n)^\infty_{n=1}$ is a Cauchy sequence.
    Then, the sequence is eventually $\varepsilon$-steady for every rational number $\varepsilon > 0$.
    In particular, this sequence is eventually $1$-steady, or the sequence $(a_n)^\infty_{n=N}$ is $1$-steady.
    The sequence $(a_n)^{N-1}_{n=1}$ is finite, and thus is bounded by Lemma 5.1.14.
    By the definition of bounded sequences, the sequence is bounded by some rational $M \ge 0$.
    
    The infinite sequence $(a_n)^\infty_{n=N}$ is eventually $1$-steady.
    This implies $d(a_j, a_k) = \abs{a_j - a_k} \le 1$ for all $j, k \ge N$.
    Take $k = N$, so $\abs{a_j - a_N} \le 1$.
    Then, 
    \[
        \abs{a_j - a_N} + \abs{a_N} \le 1 + \abs{a_N}.
    \]
    By Proposition 4.3.3(b), we have
    \[
        \abs{a_j - a_N + a_N} = \abs{a_j} \le \abs{a_j - a_N} + \abs{a_N}.
    \]
    Thus, by transitivity of inequality, we have $\abs{a_j} \le 1 + \abs{a_N}$.
    Because $1 + \abs{a_N}$ is rational, the infinite sequence $(a_n)^\infty_{n=N}$ is bounded.
    Thus, the bound of the entire sequence is $\max{(M, 1 + \abs{a_N})}$\footnote{The maximum function is as defined by Tao in Lemma 5.3.6; however, this is not circular as its definition only relies on Proposition 2.2.13.}. \qed
\end{exer}


\subsection{Equivalent Cauchy sequences}

\begin{exer}{5.2.1}{
    Show that if $(a_n)^\infty_{n=1}$ and $(b_n)^\infty_{n=1}$ are equivalent sequences of rationals, then $(a_n)^\infty_{n=1}$ is a Cauchy sequence if and only if $(b_n)^\infty_{n=1}$ is a Cauchy sequence.
}
    Suppose $(a_n)^\infty_{n=1}$ and $(b_n)^\infty_{n=1}$ are equivalent sequences of rationals.
    Suppose also $(a_n)^\infty_{n=1}$ is a Cauchy sequence; we will prove $(b_n)^\infty_{n=1}$ is a Cauchy sequence.
    The converse can be proven by reversing the roles of the sequences.
    
    From the definition of Cauchy sequences (Definition 5.1.8), there exists an $N_1 \ge 1$ such that $\abs{a_j - a_k} \le \varepsilon$ for all $j, k \ge N_1$.
    By the definition of equivalent sequences (Definition 5.2.6), there exists an $N_2 \ge 1$ such that $\abs{a_n - b_n} \le \varepsilon$ for all $n \ge N_2$.
    Take $N_3 \coloneqq \max(N_1, N_2)$, $\varepsilon \coloneqq \delta/3$, and consider $j, k \ge N_3$.
    Then, we have
    \[
        \abs{a_j - a_k} \le \delta / 3 \qquad \abs{a_k - b_k} \le \delta/3 \qquad \abs{b_j - a_j} = \abs{a_j - b_j} \le \delta /3.
    \]
    By Proposition 4.3.3(b), we have
    \begin{align*}
        \abs{b_j - b_k}
        &= \abs{(b_j - a_j) + (a_j - a_k) + (a_k - b_k)} \\
        &\le \abs{b_j - a_j} + \abs{a_j - a_k} + \abs{a_k - b_k} \\
        &\le \delta/3 + \delta/3 + \delta/3 = \delta.
    \end{align*}
    Hence, for every $\delta > 0$, there exists $N_3$ such that $\abs{b_j - b_k} \le \delta$ for any $j, k \ge N_3$.
    This implies $(b_n)^\infty_{n=1}$ is a Cauchy sequence by definition. \qed
\end{exer}


\begin{exer}{5.2.2}{
    Let $\varepsilon > 0$.
    Show that if $(a_n)^\infty_{n=1}$ and $(b_n)^\infty_{n=1}$ are eventually $\varepsilon$-close, then $(a_n)^\infty_{n=1}$ is bounded if and only if $(b_n)^\infty_{n=1}$ is bounded.
}
    Let $\varepsilon > 0$ as given.
    Suppose $(a_n)^\infty_{n=1}$ and $(b_n)^\infty_{n=1}$ are eventually $\varepsilon$-close.
    Suppose also $(a_n)^\infty_{n=1}$ is bounded; we will prove $(b_n)^\infty_{n=1}$ is bounded.
    The converse can be proven by swapping the roles of the sequences.

    By definition of eventually $\varepsilon$-close sequences, there exists an $M \ge 0$ such that $\abs{b_n - a_n} \le \varepsilon$ for all $n \ge M$.
    From the definition of bounded sequences, there exists a rational $N \ge 0$ such that $\abs{a_i} \le N$ for all $i \ge 1$.
    Take $i = n$.
    Then, by Proposition 4.3.3(b), we have
    \[
        \abs{b_n - a_n} + \abs{a_n} \ge \abs{b_n - a_n + a_n} = \abs{b_n}.
    \]
    But also,
    \[
        \abs{b_n - a_n} + \abs{a_n} \le \varepsilon + N.
    \]
    Thus, $\abs{b_n} \le \varepsilon + N$ for all $n$; it follows that $(b_n)^\infty_{n=M}$ is bounded by $(N + \varepsilon)$.
    By Lemma 5.1.14, $(b_n)^{M-1}_{n=1}$ is finite, and thus is bounded by some rational $Z \ge 0$.
    Hence, $(b_n)^\infty_{n=1}$ is bounded by $\max{(Z, \varepsilon + N)}$. \qed
\end{exer}
